5   public class InputParser implements IInputParser {
6       private String[] args;
7       int numOfClients;
8       int numOfVerifs;
9       double fraction;
10      String outputFile;
11  
12      public InputParser(String[] arguments) {
13          args = arguments;
14          numOfClients = Integer.valueOf(arguments[0]);
15          numOfVerifs = Integer.valueOf(arguments[1]);
16          fraction = Double.valueOf(arguments[2]);
17          outputFile = arguments[3];
18      }
19  
20      public void checkArgs() {
21          //when argument number is less than 4
22          if (args.length < 4) {
23              System.out.println("Error: One or more arguments are missing.");
24              throw new IllegalArgumentException("invalid number of arguments");
25          }
26  
27          //when number of clients are out of range
28          if (numOfClients <= 0 || numOfClients > 50000) {
29              System.out.println("Error: number of clients should be in range [1, 50000]");
30              throw new IllegalArgumentException("Error: number of clients should be in range [1, 50000]");
31          }
32  
33          //when number of verifications are out of range
34          if (numOfVerifs <= 0 || numOfVerifs > 10000) {
35              System.out.println("Error: number of verifications should be in range [1, 10000]");
36              throw new IllegalArgumentException("Error: number of verifications should be in range [1, 10000]");
37          }
38  
39          //when fraction are out of range
40          if (fraction < 0 || fraction > 1) {
41              System.out.println("Error: fraction should be in range [0, 1]");
42              throw new IllegalArgumentException("Error: fraction should be in range [0, 1]");
43          }
44  
45          //try to get third optional argument  numberOfRecommendations
46          if(!outputFile.endsWith(".csv")) {
47              throw  new IllegalArgumentException("Error: invalid file extension, filename should end with \".csv\"");
48          }
49      }
50  }
