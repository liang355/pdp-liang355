31          /** Compute phi(n).
32          * @param n The positive argument of the function.
33          * @return phi(n)
34          */
35          public BigInteger at(BigInteger n)
36          {
37                  if ( n.compareTo(BigInteger.ZERO) <= 0 )
38                          throw new ArithmeticException("negative argument "+n+ " of EulerPhi") ;
39                  Ifactor prFact = new Ifactor(n) ;
40                  BigInteger phi = n ;
41                  if ( n.compareTo(BigInteger.ONE) > 0 )
42                          for(int i=0 ; i < prFact.primeexp.size() ; i += 2)
43                          {
44                                  BigInteger p = new BigInteger(prFact.primeexp.elementAt(i).toString()) ;
45                                  BigInteger p_1 = p.subtract(BigInteger.ONE) ;
46                                  phi = phi.multiply(p_1).divide(p) ;
47                          }
48                  return phi ;
49          } /* at */
