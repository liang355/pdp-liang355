121         * @return 3.14159...

125         {
126                 /* look it up if possible */
127                 if ( mc.getPrecision() < PI.precision() )

134                         BigDecimal S = broadhurstBBP(1,1,a,mc) ;

139         /** Euler-Mascheroni constant.
140         * @param mc The required precision of the result.
141         * @return 0.577...
142         * @since 2009-08-13
143         */
144         static public BigDecimal gamma(MathContext mc)
145         {
146                 /* look it up if possible */
147                 if ( mc.getPrecision() < GAMMA.precision() )
148                         return GAMMA.round(mc) ;
149                 else
150                 {
151                         double eps = prec2err(0.577, mc.getPrecision() ) ;
152 
153 
154                         /* Euler-Stieltjes as shown in Dilcher, Aequat Math 48 (1) (1994) 55-85
155                         */
156                         MathContext mcloc =  new MathContext(2+mc.getPrecision()) ;
157                         BigDecimal resul =  BigDecimal.ONE ;
158                         resul =  resul.add( log(2,mcloc) ) ;
159                         resul =  resul.subtract( log(3,mcloc) ) ;
160 
161                         /* how many terms: zeta-1 falls as 1/2^(2n+1), so the
162                         * terms drop faster than 1/2^(4n+2). Set 1/2^(4kmax+2) < eps.
163                         * Leading term zeta(3)/(4^1*3) is 0.017. Leading zeta(3) is 1.2. Log(2) is 0.7
164                         */
165                         int kmax = (int)((Math.log(eps/0.7)-2.)/4.) ;
166                         mcloc =  new MathContext( 1+err2prec(1.2,eps/kmax) ) ;
167                         for(int n=1; ; n++)
168                         {
169                                 /* zeta is close to 1. Division of zeta-1 through
170                                 * 4^n*(2n+1) means divion through roughly 2^(2n+1)
171                                 */
172                                 BigDecimal c = zeta(2*n+1,mcloc).subtract(BigDecimal.ONE) ;
173                                 BigInteger fourn = new BigInteger(""+(2*n+1)) ;
174                                 fourn = fourn.shiftLeft(2*n) ;
175                                 c = divideRound(c, fourn) ;
176                                 resul = resul.subtract(c) ;
177                                 if ( c.doubleValue() < 0.1*eps)
178                                         break;
179                         }
180                         return resul.round(mc) ;
181                 }
182 
183         } /* BigDecimalMath.gamma */

186         /** The square root.
187         * @param x the non-negative argument.
188         * @param mc
189         * @return the square root of the BigDecimal.
190         * @since 2008-10-27
191         */
192         static public BigDecimal sqrt(final BigDecimal x, final MathContext mc)
193         {
194                 if ( x.compareTo(BigDecimal.ZERO) < 0 )
195                         throw new ArithmeticException("negative argument "+x.toString()+ " of square root") ;
196                 if ( x.abs().subtract( new BigDecimal(Math.pow(10.,-mc.getPrecision())) ).compareTo(BigDecimal.ZERO) < 0 )
197                         return BigDecimalMath.scalePrec(BigDecimal.ZERO,mc) ;
198                 /* start the computation from a double precision estimate */
199                 BigDecimal s = new BigDecimal( Math.sqrt(x.doubleValue()) ,mc) ;
200                 final BigDecimal half = new BigDecimal("2") ;
201 
202                 /* increase the local accuracy by 2 digits */
203                 MathContext locmc = new MathContext(mc.getPrecision()+2,mc.getRoundingMode()) ;
204 
205                 /* relative accuracy requested is 10^(-precision) 
206                 */
207                 final double eps = Math.pow(10.0,-mc.getPrecision()) ;
208                 for (;;)
209                 {
210                         /* s = s -(s/2-x/2s); test correction s-x/s for being
211                         * smaller than the precision requested. The relative correction is 1-x/s^2,
212                         * (actually half of this, which we use for a little bit of additional protection).
213                         */
214                         if ( Math.abs(BigDecimal.ONE.subtract(x.divide(s.pow(2,locmc),locmc)).doubleValue()) < eps)
215                                 break ;
216                         s = s.add(x.divide(s,locmc)).divide(half,locmc) ;
217                         /* debugging
218                         * System.out.println("itr "+x.round(locmc).toString() + " " + s.round(locmc).toString()) ;
219                         */
220                 }
221                 return s ;
222         } /* BigDecimalMath.sqrt */

251         /** The integer root.
252         * @param n the positive argument.
253         * @param x the non-negative argument.
254         * @return The n-th root of the BigDecimal rounded to the precision implied by x, x^(1/n).
255         * @since 2009-07-30
256         */
257         static public BigDecimal root(final int n, final BigDecimal x)
258         {
259                 if ( x.compareTo(BigDecimal.ZERO) < 0 )
260                         throw new ArithmeticException("negative argument "+x.toString()+ " of root") ;
261                 if ( n<= 0 )
262                         throw new ArithmeticException("negative power "+ n + " of root") ;
263 
264                 if ( n == 1 )
265                         return x ;
266 
267                 /* start the computation from a double precision estimate */
268                 BigDecimal s = new BigDecimal( Math.pow(x.doubleValue(),1.0/n) ) ;
269 
270                 /* this creates nth with nominal precision of 1 digit
271                 */
272                 final BigDecimal nth = new BigDecimal(n) ;
273 
274                 /* Specify an internal accuracy within the loop which is
275                 * slightly larger than what is demanded by 'eps' below.
276                 */
277                 final BigDecimal xhighpr = scalePrec(x,2) ;
278                 MathContext mc = new MathContext( 2+x.precision() ) ;
279 
280                 /* Relative accuracy of the result is eps.
281                 */
282                 final double eps = x.ulp().doubleValue()/(2*n*x.doubleValue()) ;
283                 for (;;)
284                 {
285                         /* s = s -(s/n-x/n/s^(n-1)) = s-(s-x/s^(n-1))/n; test correction s/n-x/s for being
286                         * smaller than the precision requested. The relative correction is (1-x/s^n)/n,
287                         */
288                         BigDecimal c = xhighpr.divide( s.pow(n-1),mc ) ;
289                         c = s.subtract(c) ;
290                         MathContext locmc = new MathContext( c.precision() ) ;
291                         c = c.divide(nth,locmc) ;
292                         s = s. subtract(c) ;
293                         if ( Math.abs( c.doubleValue()/s.doubleValue()) < eps)
294                                 break ;
295                 }
296                 return s.round(new MathContext( err2prec(eps)) ) ;
297         } /* BigDecimalMath.root */

299         /** The hypotenuse.
300         * @param x the first argument.
301         * @param y the second argument.
302         * @return the square root of the sum of the squares of the two arguments, sqrt(x^2+y^2).
303         * @since 2009-06-25
304         */
305         static public BigDecimal hypot(final BigDecimal x, final BigDecimal y)
306         {
307                 /* compute x^2+y^2
308                 */
309                 BigDecimal z = x.pow(2).add(y.pow(2)) ;
310 
311                 /* truncate to the precision set by x and y. Absolute error = 2*x*xerr+2*y*yerr,
312                 * where the two errors are 1/2 of the ulp's.  Two intermediate protectio digits.
313                 */
314                 BigDecimal zerr = x.abs().multiply(x.ulp()).add(y.abs().multiply(y.ulp())) ;
315                 MathContext mc = new MathContext(  2+err2prec(z,zerr) ) ;
316 
317                 /* Pull square root */
318                 z = sqrt(z.round(mc)) ;
319 
320                 /* Final rounding. Absolute error in the square root is (y*yerr+x*xerr)/z, where zerr holds 2*(x*xerr+y*yerr).
321                 */
322                 mc = new MathContext(  err2prec(z.doubleValue() ,0.5*zerr.doubleValue() /z.doubleValue() ) ) ;
323                 return z.round(mc) ;
324         } /* BigDecimalMath.hypot */

326         /** The hypotenuse.
327         * @param n the first argument.
328         * @param x the second argument.
329         * @return the square root of the sum of the squares of the two arguments, sqrt(n^2+x^2).
330         * @since 2009-08-05
331         */
332         static public BigDecimal hypot(final int n, final BigDecimal x)
333         {
334                 /* compute n^2+x^2 in infinite precision
335                 */
336                 BigDecimal z = (new BigDecimal(n)).pow(2).add(x.pow(2)) ;
337 
338                 /* Truncate to the precision set by x. Absolute error = in z (square of the result) is |2*x*xerr|,
339                 * where the error is 1/2 of the ulp. Two intermediate protection digits.
340                 * zerr is a signed value, but used only in conjunction with err2prec(), so this feature does not harm. 
341                 */
342                 double zerr = x.doubleValue()*x.ulp().doubleValue() ;
343                 MathContext mc = new MathContext(  2+err2prec(z.doubleValue(),zerr) ) ;
344 
345                 /* Pull square root */
346                 z = sqrt(z.round(mc)) ;
347 
348                 /* Final rounding. Absolute error in the square root is x*xerr/z, where zerr holds 2*x*xerr.
349                 */
350                 mc = new MathContext(  err2prec(z.doubleValue(),0.5*zerr/z.doubleValue() ) ) ;
351                 return z.round(mc) ;
352         } /* BigDecimalMath.hypot */

355         /** A suggestion for the maximum numter of terms in the Taylor expansion of the exponential.
356         */
357         static private int TAYLOR_NTERM = 8 ;
358 
359         /** The exponential function.
360         * @param x the argument.
361         * @return exp(x).
362         * The precision of the result is implicitly defined by the precision in the argument.
363         * In particular this means that "Invalid Operation" errors are thrown if catastrophic
364         * cancellation of digits causes the result to have no valid digits left.
365         * @since 2009-05-29
366         * @author Richard J. Mathar
367         */
368         static public BigDecimal exp(BigDecimal x)
369         {
370                 /* To calculate the value if x is negative, use exp(-x) = 1/exp(x)
371                 */
372                 if ( x.compareTo(BigDecimal.ZERO) < 0 )
373                 {
374                         final BigDecimal invx = exp(x.negate() ) ;
375                         /* Relative error in inverse of invx is the same as the relative errror in invx.
376                         * This is used to define the precision of the result.
377                         */
378                         MathContext mc = new MathContext( invx.precision() ) ;
379                         return BigDecimal.ONE.divide( invx, mc ) ;
380                 }
381                 else if ( x.compareTo(BigDecimal.ZERO) == 0 )
382                 {
383                         /* recover the valid number of digits from x.ulp(), if x hits the
384                         * zero. The x.precision() is 1 then, and does not provide this information.
385                         */
386                         return scalePrec(BigDecimal.ONE, -(int)(Math.log10( x.ulp().doubleValue() )) ) ;
387                 }
388                 else
389                 {
390                         /* Push the number in the Taylor expansion down to a small
391                         * value where TAYLOR_NTERM terms will do. If x<1, the n-th term is of the order
392                         * x^n/n!, and equal to both the absolute and relative error of the result
393                         * since the result is close to 1. The x.ulp() sets the relative and absolute error
394                         * of the result, as estimated from the first Taylor term.
395                         * We want x^TAYLOR_NTERM/TAYLOR_NTERM! < x.ulp, which is guaranteed if
396                         * x^TAYLOR_NTERM < TAYLOR_NTERM*(TAYLOR_NTERM-1)*...*x.ulp.
397                         */
398                         final double xDbl = x.doubleValue() ;
399                         final double xUlpDbl = x.ulp().doubleValue() ;
400                         if ( Math.pow(xDbl,TAYLOR_NTERM) < TAYLOR_NTERM*(TAYLOR_NTERM-1.0)*(TAYLOR_NTERM-2.0)*xUlpDbl ) 
401                         {
402                                 /* Add TAYLOR_NTERM terms of the Taylor expansion (Euler's sum formula)
403                                 */
404                                 BigDecimal resul = BigDecimal.ONE ;
405 
406                                 /* x^i */
407                                 BigDecimal xpowi = BigDecimal.ONE ;
408 
409                                 /* i factorial */
410                                 BigInteger ifac = BigInteger.ONE ;
411 
412                                 /* TAYLOR_NTERM terms to be added means we move x.ulp() to the right
413                                 * for each power of 10 in TAYLOR_NTERM, so the addition won't add noise beyond
414                                 * what's already in x.
415                                 */
416                                 MathContext mcTay = new MathContext( err2prec(1.,xUlpDbl/TAYLOR_NTERM) ) ;
417                                 for(int i=1 ; i <= TAYLOR_NTERM ; i++)
418                                 {
419                                         ifac = ifac.multiply(new BigInteger(""+i) ) ;
420                                         xpowi = xpowi.multiply(x) ;
421                                         final BigDecimal c= xpowi.divide(new BigDecimal(ifac),mcTay)  ;
422                                         resul = resul.add(c) ;
423                                         if ( Math.abs(xpowi.doubleValue()) < i && Math.abs(c.doubleValue()) < 0.5* xUlpDbl )
424                                                 break;
425                                 }
426                                 /* exp(x+deltax) = exp(x)(1+deltax) if deltax is <<1. So the relative error
427                                 * in the result equals the absolute error in the argument.
428                                 */
429                                 MathContext mc = new MathContext( err2prec(xUlpDbl/2.) ) ;
430                                 return resul.round(mc) ;
431                         }
432                         else
433                         {
434                                 /* Compute exp(x) = (exp(0.1*x))^10. Division by 10 does not lead
435                                 * to loss of accuracy.
436                                 */
437                                 int exSc = (int) ( 1.0-Math.log10( TAYLOR_NTERM*(TAYLOR_NTERM-1.0)*(TAYLOR_NTERM-2.0)*xUlpDbl
438                                                         /Math.pow(xDbl,TAYLOR_NTERM) ) / ( TAYLOR_NTERM-1.0) ) ; 
439                                 BigDecimal xby10 = x.scaleByPowerOfTen(-exSc) ;
440                                 BigDecimal expxby10 = exp(xby10) ;
441 
442                                 /* Final powering by 10 means that the relative error of the result
443                                 * is 10 times the relative error of the base (First order binomial expansion).
444                                 * This looses one digit.
445                                 */
446                                 MathContext mc = new MathContext( expxby10.precision()-exSc ) ;
447                                 /* Rescaling the powers of 10 is done in chunks of a maximum of 8 to avoid an invalid operation
448                                 * response by the BigDecimal.pow library or integer overflow.
449                                 */
450                                 while ( exSc > 0 )
451                                 {
452                                         int exsub = Math.min(8,exSc) ;
453                                         exSc -= exsub ;
454                                         MathContext mctmp = new MathContext( expxby10.precision()-exsub+2 ) ;
455                                         int pex = 1 ;
456                                         while ( exsub-- > 0 )
457                                                 pex *= 10 ;
458                                         expxby10 = expxby10.pow(pex,mctmp) ;
459                                 }
460                                 return expxby10.round(mc) ;
461                         }
462                 }
463         } /* BigDecimalMath.exp */

485         /** The natural logarithm.
486         * @param x the argument.
487         * @return ln(x).
488         * The precision of the result is implicitly defined by the precision in the argument.
489         * @since 2009-05-29
490         * @author Richard J. Mathar
491         */
492         static public BigDecimal log(BigDecimal x)
493         {
494                 /* the value is undefined if x is negative.
495                 */
496                 if ( x.compareTo(BigDecimal.ZERO) < 0 )
497                         throw new ArithmeticException("Cannot take log of negative "+ x.toString() ) ;
498                 else if ( x.compareTo(BigDecimal.ONE) == 0 )
499                 {
500                         /* log 1. = 0. */
501                         return scalePrec(BigDecimal.ZERO, x.precision()-1) ;
502                 }
503                 else if ( Math.abs(x.doubleValue()-1.0) <= 0.3 )
504                 {
505                         /* The standard Taylor series around x=1, z=0, z=x-1. Abramowitz-Stegun 4.124.
506                         * The absolute error is err(z)/(1+z) = err(x)/x.
507                         */
508                         BigDecimal z = scalePrec(x.subtract(BigDecimal.ONE),2) ;
509                         BigDecimal zpown = z ;
510                         double eps = 0.5*x.ulp().doubleValue()/Math.abs(x.doubleValue()) ;
511                         BigDecimal resul = z ;
512                         for(int k= 2;; k++)
513                         {
514                                 zpown = multiplyRound(zpown,z) ;
515                                 BigDecimal c = divideRound(zpown,k) ;
516                                 if ( k % 2 == 0)
517                                         resul = resul.subtract(c) ;
518                                 else
519                                         resul = resul.add(c) ;
520                                 if ( Math.abs(c.doubleValue()) < eps)
521                                         break;
522                         }
523                         MathContext mc = new MathContext( err2prec(resul.doubleValue(),eps) ) ;
524                         return resul.round(mc) ;
525                 }
526                 else
527                 {
528                         final double xDbl = x.doubleValue() ;
529                         final double xUlpDbl = x.ulp().doubleValue() ;
530 
531                         /* Map log(x) = log root[r](x)^r = r*log( root[r](x)) with the aim
532                         * to move roor[r](x) near to 1.2 (that is, below the 0.3 appearing above), where log(1.2) is roughly 0.2.
533                         */
534                         int r = (int) (Math.log(xDbl)/0.2) ;
535 
536                         /* Since the actual requirement is a function of the value 0.3 appearing above,
537                         * we avoid the hypothetical case of endless recurrence by ensuring that r >= 2.
538                         */
539                         r = Math.max(2,r) ;
540 
541                         /* Compute r-th root with 2 additional digits of precision
542                         */
543                         BigDecimal xhighpr = scalePrec(x,2) ;
544                         BigDecimal resul = root(r,xhighpr) ;
545                         resul = log(resul).multiply(new BigDecimal(r)) ;
546 
547                         /* error propagation: log(x+errx) = log(x)+errx/x, so the absolute error
548                         * in the result equals the relative error in the input, xUlpDbl/xDbl .
549                         */
550                         MathContext mc = new MathContext( err2prec(resul.doubleValue(),xUlpDbl/xDbl) ) ;
551                         return resul.round(mc) ;
552                 }
553         } /* BigDecimalMath.log */

555         /** The natural logarithm.
556         * @param n The main argument, a strictly positive integer.
557         * @param mc The requirements on the precision.
558         * @return ln(n).
559         * @since 2009-08-08
560         * @author Richard J. Mathar
561         */
562         static public BigDecimal log(int n, final MathContext mc)
563         {
564                 /* the value is undefined if x is negative.
565                 */
566                 if ( n <= 0 )
567                         throw new ArithmeticException("Cannot take log of negative "+ n ) ;
568                 else if ( n == 1)
569                         return BigDecimal.ZERO ;
570                 else if ( n == 2)
571                 {
572                         if ( mc.getPrecision() < LOG2.precision() )
573                                 return LOG2.round(mc) ;
574                         else
575                         {
576                                 /* Broadhurst <a href="http://arxiv.org/abs/math/9803067">arXiv:math/9803067</a>
577                                 * Error propagation: the error in log(2) is twice the error in S(2,-5,...).
578                                 */
579                                 int[] a = {2,-5,-2,-7,-2,-5,2,-3} ; 
580                                 BigDecimal S = broadhurstBBP(2,1,a, new MathContext(1+mc.getPrecision()) ) ;
581                                 S = S.multiply(new BigDecimal(8)) ;
582                                 S = sqrt(divideRound(S,3)) ;
583                                 return S.round(mc) ;
584                         }
585                 }
586                 else if ( n == 3)
587                 {
588                         /* summation of a series roughly proportional to (7/500)^k. Estimate count
589                         * of terms to estimate the precision (drop the favorable additional
590                         * 1/k here): 0.013^k <= 10^(-precision), so k*log10(0.013) <= -precision
591                         * so k>= precision/1.87.
592                         */
593                         int kmax = (int)(mc.getPrecision()/1.87) ;
594                         MathContext mcloc = new MathContext( mc.getPrecision()+ 1+(int)(Math.log10(kmax*0.693/1.098)) ) ;
595                         BigDecimal log3 = multiplyRound( log(2,mcloc),19 ) ;
596 
597                         /* log3 is roughly 1, so absolute and relative error are the same. The
598                         * result will be divided by 12, so a conservative error is the one
599                         * already found in mc
600                         */
601                         double eps = prec2err(1.098,mc.getPrecision() )/kmax ;
602                         Rational r = new Rational(7153,524288) ;
603                         Rational pk = new Rational(7153,524288) ;
604                         for(int k=1; ; k++)
605                         {
606                                 Rational tmp = pk.divide(k) ;
607                                 if ( tmp.doubleValue() < eps)
608                                         break ;
609 
610                                 /* how many digits of tmp do we need in the sum?
611                                 */
612                                 mcloc = new MathContext( err2prec(tmp.doubleValue(),eps) ) ;
613                                 BigDecimal c = pk.divide(k).BigDecimalValue(mcloc) ;
614                                 if ( k % 2 != 0)
615                                         log3 = log3.add(c) ;
616                                 else
617                                         log3 = log3.subtract(c) ;
618                                 pk = pk.multiply(r) ;
619                         }
620                         log3 = divideRound( log3,12 ) ;
621                         return log3.round(mc) ;
622                 }
623                 else if ( n == 5)
624                 {
625                         /* summation of a series roughly proportional to (7/160)^k. Estimate count
626                         * of terms to estimate the precision (drop the favorable additional
627                         * 1/k here): 0.046^k <= 10^(-precision), so k*log10(0.046) <= -precision
628                         * so k>= precision/1.33.
629                         */
630                         int kmax = (int)(mc.getPrecision()/1.33) ;
631                         MathContext mcloc = new MathContext( mc.getPrecision()+ 1+(int)(Math.log10(kmax*0.693/1.609)) ) ;
632                         BigDecimal log5 = multiplyRound( log(2,mcloc),14 ) ;
633 
634                         /* log5 is roughly 1.6, so absolute and relative error are the same. The
635                         * result will be divided by 6, so a conservative error is the one
636                         * already found in mc
637                         */
638                         double eps = prec2err(1.6,mc.getPrecision() )/kmax ;
639                         Rational r = new Rational(759,16384) ;
640                         Rational pk = new Rational(759,16384) ;
641                         for(int k=1; ; k++)
642                         {
643                                 Rational tmp = pk.divide(k) ;
644                                 if ( tmp.doubleValue() < eps)
645                                         break ;
646 
647                                 /* how many digits of tmp do we need in the sum?
648                                 */
649                                 mcloc = new MathContext( err2prec(tmp.doubleValue(),eps) ) ;
650                                 BigDecimal c = pk.divide(k).BigDecimalValue(mcloc) ;
651                                 log5 = log5.subtract(c) ;
652                                 pk = pk.multiply(r) ;
653                         }
654                         log5 = divideRound( log5,6 ) ;
655                         return log5.round(mc) ;
656                 }
657                 else if ( n == 7)
658                 {
659                         /* summation of a series roughly proportional to (1/8)^k. Estimate count
660                         * of terms to estimate the precision (drop the favorable additional
661                         * 1/k here): 0.125^k <= 10^(-precision), so k*log10(0.125) <= -precision
662                         * so k>= precision/0.903.
663                         */
664                         int kmax = (int)(mc.getPrecision()/0.903) ;
665                         MathContext mcloc = new MathContext( mc.getPrecision()+ 1+(int)(Math.log10(kmax*3*0.693/1.098)) ) ;
666                         BigDecimal log7 = multiplyRound( log(2,mcloc),3 ) ;
667 
668                         /* log7 is roughly 1.9, so absolute and relative error are the same.
669                         */
670                         double eps = prec2err(1.9,mc.getPrecision() )/kmax ;
671                         Rational r = new Rational(1,8) ;
672                         Rational pk = new Rational(1,8) ;
673                         for(int k=1; ; k++)
674                         {
675                                 Rational tmp = pk.divide(k) ;
676                                 if ( tmp.doubleValue() < eps)
677                                         break ;
678 
679                                 /* how many digits of tmp do we need in the sum?
680                                 */
681                                 mcloc = new MathContext( err2prec(tmp.doubleValue(),eps) ) ;
682                                 BigDecimal c = pk.divide(k).BigDecimalValue(mcloc) ;
683                                 log7 = log7.subtract(c) ;
684                                 pk = pk.multiply(r) ;
685                         }
686                         return log7.round(mc) ;
687 
688                 }
689 
690                 else
691                 {
692                         /* At this point one could either forward to the log(BigDecimal) signature (implemented)
693                         * or decompose n into Ifactors and use an implemenation of all the prime bases.
694                         * Estimate of the result; convert the mc argument to an  absolute error eps
695                         * log(n+errn) = log(n)+errn/n = log(n)+eps
696                         */
697                         double res = Math.log((double)n) ;
698                         double eps = prec2err(res,mc.getPrecision() ) ;
699                         /* errn = eps*n, convert absolute error in result to requirement on absolute error in input
700                         */
701                         eps *= n ;
702                         /* Convert this absolute requirement of error in n to a relative error in n
703                         */
704                         final MathContext mcloc = new MathContext( 1+err2prec((double)n,eps ) ) ;
705                         /* Padd n with a number of zeros to trigger the required accuracy in
706                         * the standard signature method
707                         */
708                         BigDecimal nb = scalePrec(new BigDecimal(n),mcloc) ;
709                         return log(nb) ;
710                 }
711         } /* log */

825         /** Raise to a fractional power and round.
826         * @param x The base.
827         *     Generally enforced to be positive, with the exception of integer exponents where
828         *     the sign is carried over according to the parity of the exponent.
829         * @param q The exponent.
830         * @return x^q.
831         * @since 2010-05-26
832         */
833         static public BigDecimal powRound(final BigDecimal x, final Rational q)
834         {
835                 /** Special cases: x^1=x and x^0 = 1
836                 */
837                 if ( q.compareTo(BigInteger.ONE) == 0 )
838                         return x;
839                 else if ( q.signum() == 0 )
840                         return BigDecimal.ONE ;
841                 else if ( q.isInteger() )
842                 {
843                         /* We are sure that the denominator is positive here, because normalize() has been
844                         * called during constrution etc.
845                         */
846                                 return powRound(x,q.a) ;
847                 }
848                         /* Refuse to operate on the general negative basis. The integer q have already been handled above.
849                         */
850                 else if ( x.compareTo(BigDecimal.ZERO) < 0 )
851                         throw new ArithmeticException("Cannot power negative "+ x.toString() ) ;
852                 else
853                 {
854                         if ( q.isIntegerFrac() )
855                         {
856                                 /* Newton method with first estimate in double precision.
857                                 * The disadvantage of this first line here is that the result must fit in the
858                                 * standard range of double precision numbers exponents.
859                                 */
860                                 double estim = Math.pow( x.doubleValue(),q.doubleValue() ) ;
861                                 BigDecimal res = new BigDecimal(estim) ;
862 
863                                 /* The error in x^q is q*x^(q-1)*Delta(x).
864                                 * The relative error is q*Delta(x)/x, q times the relative error of x.
865                                 */
866                                 BigDecimal reserr = new BigDecimal( 0.5* q.abs().doubleValue() 
867                                                                 * x.ulp().divide(x.abs(),MathContext.DECIMAL64).doubleValue() ) ;
868 
869                                 /* The main point in branching the cases above is that this conversion
870                                 * will succeed for numerator and denominator of q.
871                                 */
872                                 int qa = q.a.intValue() ;
873                                 int qb = q.b.intValue() ;
874 
875                                 /* Newton iterations. */
876                                 BigDecimal xpowa = powRound(x, qa) ;
877                                 for( ;; )
878                                 {
879                                         /* numerator and denominator of the Newton term.  The major
880                                         * disadvantage of this implementation is that the updates of the powers
881                                         * of the new estimate are done in full precision calling BigDecimal.pow(),
882                                         * which becomes slow if the denominator of q is large.
883                                         */
884                                         BigDecimal nu = res.pow(qb) .subtract(xpowa) ;
885                                         BigDecimal de = multiplyRound( res.pow(qb-1),q.b) ;
886 
887                                         /* estimated correction */
888                                         BigDecimal eps = nu.divide(de,MathContext.DECIMAL64) ;
889 
890                                         BigDecimal err = res.multiply(reserr,MathContext.DECIMAL64) ;
891                                         int precDiv = 2+err2prec(eps,err) ;
892                                         if ( precDiv <= 0 )
893                                         {
894                                                 /* The case when the precision is already reached and any precision
895                                                 * will do. */
896                                                 eps = nu.divide(de,MathContext.DECIMAL32) ;
897                                         }
898                                         else
899                                         {
900                                                 MathContext mc = new MathContext(precDiv) ;
901                                                 eps = nu.divide(de,mc) ;
902                                         }
903 
904                                         res = subtractRound(res,eps) ;
905                                         /* reached final precision if the relative error fell below reserr,
906                                         * |eps/res| < reserr
907                                         */
908                                         if ( eps.divide(res,MathContext.DECIMAL64).abs().compareTo(reserr) < 0 )
909                                         {
910                                                 /* delete the bits of extra precision kept in this
911                                                 * working copy.
912                                                 */
913                                                 MathContext mc = new MathContext(err2prec(reserr.doubleValue())) ;
914                                                 return res.round(mc) ;
915                                         }
916                                 }
917                         }
918                         else
919                         {
920                                 /* The error in x^q is q*x^(q-1)*Delta(x) + Delta(q)*x^q*log(x).
921                                 * The relative error is q/x*Delta(x) + Delta(q)*log(x). Convert q to a floating point
922                                 * number such that its relative error becomes negligible: Delta(q)/q << Delta(x)/x/log(x) .
923                                 */
924                                 int precq =  3+err2prec( (x.ulp().divide(x,MathContext.DECIMAL64)).doubleValue() 
925                                                         / Math.log(x.doubleValue()) ) ;
926                                 MathContext mc = new MathContext(precq) ;
927 
928                                 /* Perform the actual calculation as exponentiation of two floating point numbers.
929                                 */
930                                 return pow(x, q.BigDecimalValue(mc) ) ;
931                         }
932 
933 
934                 }
935         } /* BigDecimalMath.powRound */

937         /** Trigonometric sine.
938         * @param x The argument in radians.
939         * @return sin(x) in the range -1 to 1.
940         * @since 2009-06-01
941         */
942         static public BigDecimal sin(final BigDecimal x)
943         {
944                 if ( x.compareTo(BigDecimal.ZERO) < 0)
945                         return sin(x.negate()).negate() ;
946                 else if ( x.compareTo(BigDecimal.ZERO) == 0 )
947                         return BigDecimal.ZERO ;
948                 else
949                 {
950                         /* reduce modulo 2pi
951                         */
952                         BigDecimal res = mod2pi(x) ;
953                         double errpi = 0.5*Math.abs(x.ulp().doubleValue()) ;
954                         MathContext mc = new MathContext( 2+err2prec(3.14159,errpi) ) ;
955                         BigDecimal p= pi(mc) ;
956                         mc = new MathContext( x.precision() ) ;
957                         if ( res.compareTo(p) > 0 )
958                         {
959                                 /* pi<x<=2pi: sin(x)= - sin(x-pi)
960                                 */
961                                 return sin(subtractRound(res,p)) .negate() ;
962                         }
963                         else if ( res.multiply(new BigDecimal("2")).compareTo(p) > 0 )
964                         {
965                                 /* pi/2<x<=pi: sin(x)= sin(pi-x)
966                                 */
967                                 return sin(subtractRound(p,res)) ;
968                         }
969                         else
970                         {
971                                 /* for the range 0<=x<Pi/2 one could use sin(2x)=2sin(x)cos(x)
972                                 * to split this further. Here, use the sine up to pi/4 and the cosine higher up.
973                                 */
974                                 if ( res.multiply(new BigDecimal("4")).compareTo(p) > 0 )
975                                 {
976                                         /* x>pi/4: sin(x) = cos(pi/2-x)
977                                         */
978                                         return cos( subtractRound(p.divide(new BigDecimal("2")),res) ) ;
979                                 }
980                                 else
981                                 {
982                                         /* Simple Taylor expansion, sum_{i=1..infinity} (-1)^(..)res^(2i+1)/(2i+1)! */
983                                         BigDecimal resul = res ;
984 
985                                         /* x^i */
986                                         BigDecimal xpowi = res ;
987 
988                                         /* 2i+1 factorial */
989                                         BigInteger ifac = BigInteger.ONE ;
990 
991                                         /* The error in the result is set by the error in x itself.
992                                         */
993                                         double xUlpDbl = res.ulp().doubleValue() ;
994 
995                                         /* The error in the result is set by the error in x itself.
996                                         * We need at most k terms to squeeze x^(2k+1)/(2k+1)! below this value.

1027         */

1029         {
1030                 if ( x.compareTo(BigDecimal.ZERO) < 0)
1031                         return cos(x.negate());
1032                 else if ( x.compareTo(BigDecimal.ZERO) == 0 )
1033                         return BigDecimal.ONE ;
1034                 else
1035                 {
1036                         /* reduce modulo 2pi
1037                         */

1062                                 {

1074 
1075                                         /* 2i factorial */
1076                                         BigInteger ifac = BigInteger.ONE ;
1077 
1078                                         /* The absolute error in the result is the error in x^2/2 which is x times the error in x.

1088                                         for(int i=1 ; ; i++)
1089                                         {

1091                                                 */

1093                                                 ifac = ifac.multiply( new BigInteger(""+(2*i)) ) ;
1094                                                 xpowi = xpowi.multiply(res).multiply(res).negate() ;

1098                                                         break ;

1101                                         */
1102                                         mc = new MathContext( err2prec(resul.doubleValue(),xUlpDbl) ) ;
1103                                         return resul.round(mc) ;
1104                                 }
1105                         }

1532         * @param x the argument.
1533         * @return the sinh(x) = (exp(x)-exp(-x))/2 .
1534         * @author Richard J. Mathar
1535         * @since 2009-08-19
1536         */
1537         static public BigDecimal sinh(final BigDecimal x)
1538         {
1539                 if ( x.compareTo(BigDecimal.ZERO) < 0)
1540                         return sinh(x.negate()).negate() ;
1541                 else if ( x.compareTo(BigDecimal.ZERO) == 0 )
1542                         return BigDecimal.ZERO ;
1543                 else
1544                 {
1545                         if ( x.doubleValue() > 2.4 )
1546                         {
1547                                 /* Move closer to zero with sinh(2x)= 2*sinh(x)*cosh(x).
1548                                 */

1550                                 BigDecimal xhalf = x.divide(two) ;
1551                                 BigDecimal resul =  sinh(xhalf).multiply(cosh(xhalf)).multiply(two) ;

1553                                 * The first derivative of sinh(x) is cosh(x), so the absolute error

1555                                 */

1558                                 return resul.round(mc) ;
1559                         }
1560                         else
1561                         {
1562                                 BigDecimal xhighpr = scalePrec(x,2) ;

1564                                 BigDecimal resul = xhighpr ;
1565 
1566                                 /* x^i */
1567                                 BigDecimal xpowi = xhighpr ;
1568 
1569                                 /* 2i+1 factorial */
1570                                 BigInteger ifac = BigInteger.ONE ;
1571 
1572                                 /* The error in the result is set by the error in x itself.
1573                                 */
1574                                 double xUlpDbl = x.ulp().doubleValue() ;

1577                                 * We need at most k terms to squeeze x^(2k+1)/(2k+1)! below this value.
1578                                 * x^(2k+1) < x.ulp; (2k+1)*log10(x) < -x.precision; 2k*log10(x)< -x.precision;
1579                                 * 2k*(-log10(x)) > x.precision; 2k*log10(1/x) > x.precision
1580                                 */
1581                                 int k = (int)(x.precision()/Math.log10(1.0/xhighpr.doubleValue()))/2 ;

1584                                 {
1585                                         /* TBD: at which precision will 2*i or 2*i+1 overflow? 
1586                                         */

1588                                         ifac = ifac.multiply( new BigInteger(""+(2*i+1)) ) ;

1592                                         if ( corr.abs().doubleValue() < 0.5*xUlpDbl ) 
1593                                                 break ;
1594                                 }
1595                                 /* The error in the result is set by the error in x itself.
1596                                 */
1597                                 MathContext mc = new MathContext(x.precision() ) ;
1598                                 return resul.round(mc) ;
1599                         }
1600                 }

1687         /** The Gamma function.
1688         * @param x The argument.
1689         * @return Gamma(x).
1690         * @since 2009-08-06
1691         */
1692         static public BigDecimal Gamma(final BigDecimal x)
1693         {
1694                 /* reduce to interval near 1.0 with the functional relation, Abramowitz-Stegun 6.1.33
1695                 */
1696                 if ( x.compareTo(BigDecimal.ZERO) < 0 )
1697                         return divideRound(Gamma( x.add(BigDecimal.ONE) ),x) ;
1698                 else if ( x.doubleValue() > 1.5 )
1699                 {
1700                         /* Gamma(x) = Gamma(xmin+n) = Gamma(xmin)*Pochhammer(xmin,n).
1701                         */
1702                         int n = (int) ( x.doubleValue()-0.5 );
1703                         BigDecimal xmin1 = x.subtract(new BigDecimal(n)) ;
1704                         return multiplyRound(Gamma(xmin1), pochhammer(xmin1,n) ) ;
1705                 }
1706                 else
1707                 {
1708                         /* apply Abramowitz-Stegun 6.1.33
1709                         */
1710                         BigDecimal z = x.subtract(BigDecimal.ONE) ;
1711 
1712                         /* add intermediately 2 digits to the partial sum accumulation
1713                         */
1714                         z = scalePrec(z,2) ;
1715                         MathContext mcloc = new MathContext(z.precision()) ;
1716 
1717                         /* measure of the absolute error is the relative error in the first, logarithmic term
1718                         */
1719                         double eps = x.ulp().doubleValue()/x.doubleValue() ;
1720 
1721                         BigDecimal resul = log( scalePrec(x,2)).negate() ;
1722 
1723                         if ( x.compareTo(BigDecimal.ONE) != 0 )
1724                         {
1725 
1726                                 BigDecimal gammCompl = BigDecimal.ONE.subtract(gamma(mcloc) ) ;
1727                                 resul = resul.add( multiplyRound(z,gammCompl) ) ;
1728                                 for(int n=2; ;n++)
1729                                 {
1730                                         /* multiplying z^n/n by zeta(n-1) means that the two relative errors add.
1731                                         * so the requirement in the relative error of zeta(n)-1 is that this is somewhat
1732                                         * smaller than the relative error in z^n/n (the absolute error of thelatter  is the
1733                                         * absolute error in z) 
1734                                         */
1735                                         BigDecimal c = divideRound(z.pow(n,mcloc),n) ;
1736                                         MathContext m = new MathContext( err2prec(n*z.ulp().doubleValue()/2./z.doubleValue()) ) ;
1737                                         c = c.round(m) ;
1738 
1739                                         /* At larger n, zeta(n)-1 is roughly 1/2^n. The product is c/2^n.
1740                                         * The relative error in c is c.ulp/2/c . The error in the product should be small versus eps/10.
1741                                         * Error from 1/2^n is c*err(sigma-1).
1742                                         * We need a relative error of zeta-1 of the order of c.ulp/50/c. This is an absolute
1743                                         * error in zeta-1 of c.ulp/50/c/2^n, and also the absolute error in zeta, because zeta is
1744                                         * of the order of 1.
1745                                         */
1746                                         if ( eps/100./c.doubleValue() < 0.01 )
1747                                                 m = new MathContext( err2prec(eps/100./c.doubleValue()) ) ;
1748                                         else
1749                                                 m = new MathContext( 2) ;
1750                                         /* zeta(n) -1 */
1751                                         BigDecimal zetm1 = zeta(n,m).subtract(BigDecimal.ONE) ;
1752                                         c = multiplyRound(c,zetm1) ;
1753 
1754                                         if ( n % 2 == 0 )
1755                                                 resul = resul.add(c) ;
1756                                         else
1757                                                 resul = resul.subtract(c) ;
1758         
1759                                         /* alternating sum, so truncating as eps is reached suffices 
1760                                         */
1761                                         if ( Math.abs(c.doubleValue()) < eps)
1762                                                 break;
1763                                 }
1764                         }
1765                         
1766                         /* The relative error in the result is the absolute error in the
1767                         * input variable times the digamma (psi) value at that point.
1768                         */
1769                         double zdbl = z.doubleValue() ;
1770                         eps = psi(zdbl)* x.ulp().doubleValue()/2. ;
1771                         mcloc = new MathContext( err2prec(eps) ) ;
1772                         return exp(resul).round(mcloc) ;
1773                 }
1774         } /* BigDecimalMath.gamma */

1776         /** The Gamma function.
1777         * @param q The argument.
1778         * @param mc The required accuracy in the result.
1779         * @return Gamma(x).
1780         * @since 2010-05-26
1781         */
1782         static public BigDecimal Gamma(final Rational q, final MathContext mc)
1783         {
1784                 if ( q.isBigInteger() )
1785                 {
1786                         if ( q.compareTo(Rational.ZERO) <= 0 )
1787                                 throw new ArithmeticException("Gamma at "+q.toString() ) ;
1788                         else
1789                         {
1790                                 /* Gamma(n) = (n-1)! */
1791                                 Factorial f = new Factorial() ;
1792                                 BigInteger g = f.at( q.trunc().intValue()-1 ) ;
1793                                 return scalePrec(new BigDecimal(g),mc) ;
1794                         }
1795                 }
1796                 else if ( q.b.intValue() == 2 )
1797                 {
1798                         /* half integer cases which are related to sqrt(pi)
1799                         */
1800                         BigDecimal p = sqrt(pi(mc)) ;
1801                         if ( q.compareTo(Rational.ZERO) >= 0 )
1802                         {
1803                                 Rational pro = Rational.ONE ;
1804                                 Rational f = q.subtract(1) ;
1805                                 while ( f.compareTo(Rational.ZERO) > 0 )
1806                                 {
1807                                         pro = pro.multiply(f) ;
1808                                         f = f.subtract(1) ;
1809                                 }
1810                                 return multiplyRound(p,pro) ;
1811                         }
1812                         else
1813                         {
1814                                 Rational pro = Rational.ONE ;
1815                                 Rational f = q ;
1816                                 while ( f.compareTo(Rational.ZERO) < 0 )
1817                                 {
1818                                         pro = pro.divide(f) ;
1819                                         f = f.add(1) ;
1820                                 }
1821                                 return multiplyRound(p,pro) ;
1822                         }
1823                 }
1824                 else
1825                 {
1826                         /* The relative error of the result is psi(x)*Delta(x). Tune Delta(x) such
1827                         * that this is equivalent to mc: Delta(x) = precision/psi(x).
1828                         */
1829                         double qdbl = q.doubleValue() ;
1830                         double deltx = 5.*Math.pow(10.,-mc.getPrecision()) /psi(qdbl) ;
1831 
1832                         MathContext mcx  = new MathContext( err2prec(qdbl,deltx) ) ;

1834

1952         /** Riemann zeta function.
1953         * @param n The positive integer argument.
1954         * @param mc Specification of the accuracy of the result.
1955         * @return zeta(n).
1956         * @since 2009-08-05
1957         */
1958         static public BigDecimal zeta(final int n, final MathContext mc)
1959         {
1960                 if( n <= 0 )
1961                         throw new ProviderException("Not implemented: zeta at negative argument "+n) ;
1962                 if( n == 1 )
1963                         throw new ArithmeticException("Pole at zeta(1) ") ;
1964 
1965                 if( n % 2 == 0 )
1966                 {
1967                         /* Even indices. Abramowitz-Stegun 23.2.16. Start with 2^(n-1)*B(n)/n!
1968                         */
1969                         Rational b = (new Bernoulli()).at(n).abs() ;
1970                         b = b.divide((new Factorial()).at(n)) ;
1971                         b = b.multiply( BigInteger.ONE.shiftLeft(n-1) );
1972 
1973                         /* to be multiplied by pi^n. Absolute error in the result of pi^n is n times
1974                         * error in pi times pi^(n-1). Relative error is n*error(pi)/pi, requested by mc.
1975                         * Need one more digit in pi if n=10, two digits if n=100 etc, and add one extra digit.
1976                         */
1977                         MathContext mcpi = new MathContext( mc.getPrecision() + (int)(Math.log10(10.0*n)) ) ;
1978                         final BigDecimal piton = pi(mcpi).pow(n,mc) ;
1979                         return multiplyRound( piton, b) ;
1980                 }
1981                 else if ( n == 3)
1982                 {
1983                         /* Broadhurst BBP <a href="http://arxiv.org/abs/math/9803067">arXiv:math/9803067</a>
1984                         * Error propagation: S31 is roughly 0.087, S33 roughly 0.131
1985                         */
1986                         int[] a31 = {1,-7,-1,10,-1,-7,1,0} ; 
1987                         int[] a33 = {1,1,-1,-2,-1,1,1,0} ; 
1988                         BigDecimal S31 = broadhurstBBP(3,1,a31,mc) ;
1989                         BigDecimal S33 = broadhurstBBP(3,3,a33,mc) ;
1990                         S31 = S31.multiply(new BigDecimal(48)) ;
1991                         S33 = S33.multiply(new BigDecimal(32)) ;
1992                         return S31.add(S33).divide(new BigDecimal(7),mc) ;
1993                 }
1994                 else if ( n == 5)
1995                 {
1996                         /* Broadhurst BBP <a href=http://arxiv.org/abs/math/9803067">arXiv:math/9803067</a>
1997                         * Error propagation: S51 is roughly -11.15, S53 roughly 22.165, S55 is roughly 0.031
1998                         * 9*2048*S51/6265 = -3.28. 7*2038*S53/61651= 5.07. 738*2048*S55/61651= 0.747.
1999                         * The result is of the order 1.03, so we add 2 digits to S51 and S52 and one digit to S55.
2000                         */
2001                         int[] a51 = {31,-1614,-31,-6212,-31,-1614,31,74552} ; 
2002                         int[] a53 = {173,284,-173,-457,-173,284,173,-111} ; 
2003                         int[] a55 = {1,0,-1,-1,-1,0,1,1} ; 
2004                         BigDecimal S51 = broadhurstBBP(5,1,a51, new MathContext(2+mc.getPrecision()) ) ;
2005                         BigDecimal S53 = broadhurstBBP(5,3,a53, new MathContext(2+mc.getPrecision()) ) ;
2006                         BigDecimal S55 = broadhurstBBP(5,5,a55, new MathContext(1+mc.getPrecision()) ) ;
2007                         S51 = S51.multiply(new BigDecimal(18432)) ;
2008                         S53 = S53.multiply(new BigDecimal(14336)) ;
2009                         S55 = S55.multiply(new BigDecimal(1511424)) ;
2010                         return S51.add(S53).subtract(S55).divide(new BigDecimal(62651),mc) ;
2011                 }
2012                 else
2013                 {
2014                         /* Cohen et al Exp Math 1 (1) (1992) 25
2015                         */
2016                         Rational betsum = new Rational() ;
2017                         Bernoulli bern = new Bernoulli() ;
2018                         Factorial fact = new Factorial() ;
2019                         for(int npr=0 ; npr <= (n+1)/2 ; npr++)
2020                         {
2021                                 Rational b = bern.at(2*npr).multiply(bern.at(n+1-2*npr)) ;
2022                                 b = b.divide(fact.at(2*npr)).divide(fact.at(n+1-2*npr)) ;
2023                                 b = b.multiply(1-2*npr) ;
2024                                 if ( npr % 2 ==0 )
2025                                         betsum = betsum.add(b) ;
2026                                 else
2027                                         betsum = betsum.subtract(b) ;
2028                         }
2029                         betsum = betsum.divide(n-1) ;
2030                         /* The first term, including the facor (2pi)^n, is essentially most
2031                         * of the result, near one. The second term below is roughly in the range 0.003 to 0.009.
2032                         * So the precision here is matching the precisionn requested by mc, and the precision
2033                         * requested for 2*pi is in absolute terms adjusted.
2034                         */

2531         /** Build the inverse and maintain the approximate accuracy.
2532         * @param z The denominator
2533         * @return The divided 1/z = [Re(z)-i*Im(z)]/ [Re^2 z + Im^2 z]
2534         * @since 2010-07-19
2535         */
2536         static public BigComplex invertRound(final BigComplex z)
2537         {
2538                 if (z.im.compareTo(BigDecimal.ZERO) == 0)
2539                 {
2540                         /* In this case with vanishing Im(x), the result is  simply 1/Re z.
2541                         */
2542                         final MathContext mc = new MathContext( z.re.precision() ) ;
2543                         return new BigComplex( BigDecimal.ONE.divide( z.re, mc) ) ;
2544                 }
2545                 else if (z.re.compareTo(BigDecimal.ZERO) == 0)
2546                 {
2547                         /* In this case with vanishing Re(z), the result is  simply -i/Im z
2548                         */
2549                         final MathContext mc = new MathContext( z.im.precision() ) ;
2550                         return new BigComplex(BigDecimal.ZERO, BigDecimal.ONE.divide( z.im, mc).negate() ) ;
2551                 }
2552                 else 
2553                 {
2554                         /* 1/(x.re+I*x.im) = 1/(x.re+x.im^2/x.re) - I /(x.im +x.re^2/x.im)
2555                         */
2556                         BigDecimal R  = addRound(z.re, divideRound(multiplyRound(z.im,z.im), z.re) ) ;
2557                         BigDecimal I  = addRound(z.im, divideRound(multiplyRound(z.re,z.re), z.im) ) ;
2558                         MathContext mc = new MathContext( 1+R.precision() ) ;
2559                         R = BigDecimal.ONE.divide(R,mc) ;
2560                         mc = new MathContext( 1+I.precision() ) ;
2561                         I = BigDecimal.ONE.divide(I,mc) ;
2562                         return new BigComplex(R,I.negate()) ;
2563                 }
2564         }

2706         * @param x The value of the variable
2707         *    The value returned depends only on the absolute value, not on the sign.
2708         * @param xerr The absolute error in the variable
2709         *    The value returned depends only on the absolute value, not on the sign.
2710         * @return The number of valid digits in x.
2711         *    Derived from the representation x+- xerr, as if the error was represented
