1   package org.nevec.rjm ;

3   import java.util.* ;
4   import java.math.* ;

6   
7   /** Factorials.
8   * @since 2006-06-25
9   * @since 2012-02-15 Storage of the values based on Ifactor, not BigInteger.
10  * @author Richard J. Mathar
11  */
12  public class Factorial
13  {
14          /** The list of all factorials as a vector.
15          */
16          static Vector<Ifactor> a = new Vector<Ifactor>() ;
17  
18          /** ctor().
19          * Initialize the vector of the factorials with 0!=1 and 1!=1.
20          */
21          public Factorial()
22          {
23                  if ( a.size() == 0 )
24                  {
25                          a.add(Ifactor.ONE) ;
26                          a.add(Ifactor.ONE) ;
27                  }
28          } /* ctor */
29  
30          /** Compute the factorial of the non-negative integer.
31          * @param n the argument to the factorial, non-negative.
32          * @return the factorial of n.
33          */
34          public BigInteger at(int n)
35          {
36                  /* extend the internal list if needed.
37                  */
38                  growto(n) ;
39                  return a.elementAt(n).n ;
40          } /* at */
41  
42          /** Compute the factorial of the non-negative integer.
43          * @param n the argument to the factorial, non-negative.
44          * @return the factorial of n.
45          */
46          public Ifactor toIfactor(int n)
47          {
48                  /* extend the internal list if needed.
49                  */
50                  growto(n) ;
51                  return a.elementAt(n) ;
52          } /* at */
53  
54          /** Extend the internal table to cover up to n!
55          * @param n The maximum factorial to be supported.
56          * @since 2012-02-15
57          */
58          private void growto(int n)
59          {
60                  /* extend the internal list if needed. Size to be 2 for n<=1, 3 for n<=2 etc.
61                  */
62                  while ( a.size() <=n )
63                  {
64                          final int lastn = a.size()-1 ;
65                          final Ifactor nextn = new Ifactor(lastn+1) ;
66                          a.add(a.elementAt(lastn).multiply(nextn) ) ;
67                  }
68          } /* growto */
69  
70  } /* Factorial */
