8   /** A one-parameter polynomial with rational coefficients.

72          /** Constructor from a hypergeometric series.
73          * @param A the list of values in the numerator of AFB.
74          *   At least one of these values must be a negative integer, which implicitly determines
75          *   the order of the new polynomial.
76          * @param B the list of values in the denominator of AFB
77          * @since 2009-08-05
78          */
79          public RatPoly(final Vector<BigInteger> A, final Vector<BigInteger> B)
80          {
81                  BigInteger Nmax = BigInteger.ONE.negate() ;
82                  for(int j=0; j < A.size() ; j++)
83                  {
84                          if ( A.elementAt(j).compareTo(BigInteger.ZERO) <= 0)
85                          {
86                                  if  ( Nmax.compareTo(BigInteger.ZERO) < 0 )
87                                          Nmax = A.elementAt(j).negate() ;
88                                  else
89                                          Nmax = Nmax.min( A.elementAt(j).negate() ) ;
90                          }
91                  }
92                  if ( Nmax.compareTo(BigInteger.ZERO) < 0 )
93                          throw new ArithmeticException("Infinite Number of Terms in Series "+Nmax.toString()) ;
94  
95                  int nmax = Nmax.intValue()-1 ;
96                  init(A,B,nmax) ;
97          } /* ctor */

99          /** Constructor from a hypergeometric series.
100         * @param A the list of values in the numerator of AFB
101         * @param B the list of values in the denominator of AFB
102         * @param nmax the order of the truncated polynomial representation
103         * @since 2008-11-13
104         */
105         protected void init(final Vector<BigInteger> A, final Vector<BigInteger> B, int nmax)
106         {
107                 a = new Vector<Rational>() ;
108                 Factorial f=new Factorial() ;
109                 for( int n=0; n <= nmax ; n++)
110                 {
111                         Rational c = new Rational(1,1) ;
112                         for(int j=0; j < A.size() ; j++)
113                         {
114                                 Rational aEl = new Rational(A.elementAt(j)) ;
115                                 c = c.multiply(aEl.Pochhammer(n)) ;
116                         }
117                         for(int j=0; j < B.size() ; j++)
118                         {
119                                 Rational bEl = new Rational(B.elementAt(j)) ;
120                                 c = c.divide(bEl.Pochhammer(n)) ;
121                         }
122                         c =c.divide(f.at(n)) ;
123                         a.add(c) ;
124                 }
125                 simplify() ;
126         } /* init */

320         /** Multiply by another polynomial

325         {
326                 RatPoly resul = new RatPoly() ;
327                 /* the degree of the result is the sum of the two degrees.
328                 */
329                 final int nmax = degree()+val.degree() ;
330                 for(int n=0; n <= nmax ; n++)

334                         {

336                         }
337                         resul.set(n,coef) ;
338                 }
339                 resul.simplify() ;

343         /** Raise to a positive power.
344         * @param n The non-negative exponent of the power
345         * @return The n-th power of this.
346         */
347         public RatPoly pow(final int n) throws ArithmeticException
348         {
349                 RatPoly resul = new RatPoly("1") ;
350                 if ( n < 0 )
351                         throw new ArithmeticException("negative polynomial power "+n) ;
352                 else
353                 {
354                         /* this ought probably be done with some binary representation
355                         * of the power and a smaller number of multiplications.
356                         */
357                         for(int i=1 ; i <= n ; i++)
358                                 resul = resul.multiply(this) ;
359                         resul.simplify() ;
360                         return resul ;
361                 }
362         } /* pow */

364         /** Raise to a rational power.
365         * The result is the taylor expansion of this, truncated at the first
366         * term that remains undetermined based on the current number of coefficients.
367         * @param r the exponent of the power
368         * @return This^r .
369         * @since 2009-05-18
370         */
371         public RatPoly pow(final Rational r) throws ArithmeticException
372         {
373                 /* split (a0+a1*x+a2*x^2+...)^r = a0^r*(1+a1/a0*r+a2/a0*r^2+..)^r
374                 */
375                 Rational f = at(0) ;
376                 f = f.pow(r) ;
377 
378                 /* scale the polynomial by division through the expansion coefficient of the absolute term
379                 */
380                 RatPoly red = divide(a.elementAt(0)) ;
381 
382                 /* and remove the leading term (now equal to 1)
383                 */
384                 red.set(0,0) ;
385 
386                 /* Binomial expansion of the rest. sum_{l=0..infinity} binomial(r,l)*red^l
387                 */
388                 RatPoly resul = new RatPoly("1") ;
389 
390                 final int d = degree() ;
391                 for(int l=1 ; l <= d ; l++)
392                 {
393                         final Rational b = Rational.binomial(r,l) ;
394                         resul = resul.add( red.pow(l).multiply(b) ) ;
395                 }
396                 return resul.multiply(f) ;
397         } /* pow */

457         /** Divide by another polynomial.
458         * @param val the other polynomial
459         * @param nmax the maximum degree of the Taylor expansion of the result.
460         * @return the Taylor expansion of this/val up to degree nmax.
461         */
462         public RatPoly divide(final RatPoly val,int nmax)
463         {
464                 RatPoly num = this ;
465                 RatPoly denom = val ;
466 
467                 /* divide by a common smallest power/degree
468                 */
469                 while( num.at(0).compareTo(BigInteger.ZERO) == 0 && denom.at(0).compareTo(BigInteger.ZERO) == 0)
470                 {
471                         num.a.remove(0) ;
472                         denom.a.remove(0) ;
473                         if( num.size() <= 1 || denom.size() <= 1)
474                                 break ;
475                 }
476 
477                 RatPoly resul = new RatPoly() ;
478                 /* todo: If the polynomial division is exact, we could leave
479                 * the loop earlier, indeed
480                 */
481                 for(int n=0; n <= nmax ; n++)
482                 {
483                         Rational coef = num.at(n) ;
484                         for(int nres=0; nres < n ; nres++)
485                         {
486                                 coef = coef.subtract(resul.at(nres).multiply(denom.at(n-nres))) ;
487                         }
488                         coef = coef.divide(denom.at(0)) ;
489                         resul.set(n,coef) ;
490                 }
491                 resul.simplify() ;
492                 return(resul) ;
493         } /* divide */

495         /** Divide by another polynomial.
496         * @param val the other polynomial
497         * @return A vector with [0] containg the polynomial of degree which is the
498         *  difference of thisdegree and the degree of val. [1] the remainder polynomial.
499         * This = returnvalue[0] + returnvalue[1]/val .
500         * @since 2012-03-01
501         */
502         public RatPoly[] divideAndRemainder(final RatPoly val)
503         {
504                 RatPoly[] ret = new RatPoly[2] ;
505                 /* remove any high-order zeros
506                 */
507                 RatPoly valSimpl = val.clone() ;
508                 valSimpl.simplify() ;
509                 RatPoly thisSimpl = clone() ;
510                 thisSimpl.simplify() ;
511 
512                 /* catch the case with val equal to zero
513                 */
514                 if ( valSimpl.degree() == 0 && valSimpl.a.firstElement().compareTo(Rational.ZERO) == 0)
515                         throw new ArithmeticException("Division through zero polynomial") ;
516                 /* degree of this smaller than degree of val: remainder is this
517                 */
518                 if ( thisSimpl.degree() < valSimpl.degree() )
519                 {
520                         /* leading polynomial equals zero 
521                         */
522                         ret[0] = new RatPoly() ;
523                         ret[1] = thisSimpl ;
524                 }
525                 else
526                 {
527                         /* long division. Highest degree by dividing the highest degree
528                         * of this thru val.
529                         */
530                         ret[0] = new RatPoly() ;
531                         ret[0].set(thisSimpl.degree()-valSimpl.degree(),
532                                 thisSimpl.a.lastElement().divide(valSimpl.a.lastElement()) ) ;
533 
534                         /* recurrences: build this - val*(1-termresult) and feed this
535                         * into another round of division. Have intermediate ret[0]+ret[1]/val.
536                         */
537                         ret[1] = thisSimpl.subtract( ret[0].multiply( valSimpl) );
538 
539                         /* any remainder left ?
540                         */
541                         if ( ret[1].degree() < valSimpl.degree() )
542                                 ;
543                         else
544                         {
545                                 RatPoly rem[] = ret[1].divideAndRemainder(val) ;
546                                 ret[0] = ret[0].add(rem[0]) ;
547                                 ret[1] = rem[1] ;
548                         }
549                 }
550                 return ret ;
551         } /* divideAndRemainder */

607         /** Simplify the representation.
608         * Trailing values with zero coefficients (at high powers) are deleted.
609         * This modifies the polynomial on the stop (does not return another instance)
610         */
611         private void simplify()
612         {
613                 int n = a.size()-1 ;
614                 if ( n >= 0)
615                         while( a.elementAt(n).compareTo(BigInteger.ZERO) == 0 )
616                         {
617                                 a.remove(n) ;
618                                 if( --n <0)
619                                         break ;
620                         }
621         } /* simplify */

662         /** Mobius transform.
663         * @param maxdeg the maximum polynomial degree of the result
664         * @return the sequence of coefficients is the Mobius transform of the original sequence.
665         * @since 2008-12-02
666         */
667         public RatPoly mobiusT(int maxdeg)
668         {
669                 /* Start with the polynomial 0
670                 */
671                 RatPoly r = new RatPoly() ;
672                 for(int i=1; i <= maxdeg; i++)
673                 {
674                         Rational c = new Rational() ;
675                         for(int d=1; d <= i && d < a.size(); d++)
676                         {
677                                 if (i % d == 0) 
678                                 {
679                                         final Ifactor m = new Ifactor(i/d) ;
680                                         c = c.add( a.elementAt(d).multiply( m.moebius() ) ) ;
681                                 }
682                         }
683                         r.set(i,c) ;
684                 }
685                 r.simplify() ;
686                 return r ;
687         } /* mobiusT */

700                 {
701                         Rational c = new Rational() ;
702                         for(int d=1; d <= i && d < a.size(); d++)
703                         {
704                                 if (i % d == 0) 
705                                         c = c.add( a.elementAt(d) ) ;
706                         }
707                         r.set(i,c) ;
708                 }
709                 r.simplify() ;
710                 return r ;
711         } /* mobiusTInv */

713         /** Binomial transform.
714         * @param maxdeg the maximum polynomial degree of the result
715         * @return the sequence of coefficients is the binomial transform of the original sequence.
716         * @since 2008-10-26
717         */
718         public RatPoly binomialT(int maxdeg)
719         {
720                 RatPoly r = new RatPoly() ;
721                 for(int i=0; i <= maxdeg; i++)
722                 {
723                         Rational c = new Rational(0,1) ;
724                         for(int j=0; j <= i && j < a.size(); j++)
725                                 c = c.add( a.elementAt(j).multiply(BigIntegerMath.binomial(i,j)) ) ;
726                         r.set(i,c) ;
727                 }
728                 r.simplify() ;
729                 return r ;

733         * @param maxdeg the maximum polynomial degree of the result
734         * @return the sequence of coefficients is the inverse binomial transform of the original sequence.
735         * @since 2008-10-26
736         */
737         public RatPoly binomialTInv(int maxdeg)
738         {
739                 RatPoly r = new RatPoly() ;
740                 for(int i=0; i <= maxdeg; i++)
741                 {
742                         Rational c = new Rational(0,1) ;
743                         for(int j=0; j <= i && j < a.size(); j++)
744                                 if ( (j+i) % 2 != 0 )
745                                         c = c.subtract( a.elementAt(j).multiply(BigIntegerMath.binomial(i,j)) ) ;
746                                 else
747                                         c = c.add( a.elementAt(j).multiply(BigIntegerMath.binomial(i,j)) ) ;
748                         r.set(i,c) ;
749                 }
750                 r.simplify() ;
751                 return r ;

756         * @param newdeg The maximum degree of the result.
757         * @return The polynomial with all coefficients beyond deg set to zero.
758         * If newdeg =3, for example the polynomial returned has at most degree 3.
759         * If newdeg is larger than the degree of this, zeros (at the higher orders of x)

772         /** Generate the roots of the polynomial in floating point arithmetic.
773         * @see <a href="http://en.wikipedia.org/wiki/Durand-Kerner_method">Durand Kerner method</a>
774         * @param the number of floating point digits
775         * @since 2008-10-26
776         */
777         public Vector<BigComplex> roots(int digits)
778         {
779                 RatPoly mon = monic() ;
780 
781                 Random rand = new Random() ;
782                 MathContext mc = new MathContext(digits+3,RoundingMode.DOWN) ;
783 
784                 Vector<BigComplex> res =new Vector<BigComplex>() ;
785 
786                 final int d = mon.degree() ;
787                 double randRad =0. ;
788                 for(int i=0 ; i <= d ; i++)
789                 {
790                         /* scale coefficient at maximum degree */
791                         double absi = Math.abs( mon.at(i).doubleValue() ) ;
792                         if ( absi > randRad) 
793                                 randRad = absi ;
794                 }
795                 randRad += 1.0 ;
796 
797                 /* initial values randomly in radius 1+randRad
798                 */
799                 for(int i=0 ; i < d ; i++)
800                 {
801                         double rad = randRad*rand.nextDouble() ;
802                         double phi = 2.0*3.14159*rand.nextDouble() ;
803                         res.add(i, new BigComplex(rad*Math.cos(phi),rad*Math.sin(phi)) ) ;
804                 }
805 
806                 /* iterate until convr indicates that all values changed by less than the digits 
807                 * precision indicates.
808                 */
809                 boolean convr = false ;
810                 for(int itr =0 ; ! convr ; itr++)
811                 {
812                         convr = true ;
813                         Vector<BigComplex> resPlus =new Vector<BigComplex>() ;
814                         for(int v=0 ; v < d; v++)
815                         {
816                                 /* evaluate f(x)/(x-root1)/(x-root2)/... (x-rootdegr), Newton method
817                                 */
818                                 BigComplex thisx = res.elementAt(v) ;
819                                 BigComplex nv = mon.valueOf(thisx,mc) ;
820                                 for(int j=0; j < d ; j++)
821                                 {
822                                         if ( j != v )
823                                                 nv = nv.divide(thisx.subtract(res.elementAt(j)),mc) ;
824                                 }
825 
826                                 /* is this value converged ? */
827                                 if ( nv.abs(mc).doubleValue() >  thisx.abs(mc).doubleValue()*Math.pow(10.0,-digits) )
828                                         convr =false;
829         
830                                 thisx = thisx.subtract(nv) ;
831 
832                                 /* If unstable, start over */
833                                 if ( thisx.abs(MathContext.DECIMAL32).doubleValue() > randRad )
834                                         return roots(digits) ;
835 
836                                 resPlus.add(thisx) ;
837                         }
838                         res = resPlus ;
839 
840                 }
841                 return res;
842         } /* roots */

844         /** Generate the integer roots of the polynomial.
845         * @return The vector of integer roots, with multiplicity.
846         * The shows alternatingly first a root then its multiplicty, then another root and multiplicty etc.
847         * @since 2008-10-26
848         */
849         public Vector<BigInteger> iroots()
850         {
851                 /* The vector of the roots */
852                 Vector<BigInteger> res =new Vector<BigInteger>() ;
853 
854                 int lowd = ldegree() ;
855                 if( lowd == 0 && a.elementAt(0).compareTo(BigInteger.ZERO) == 0)
856                 {
857                         /* Case of polynomial identical to zero:
858                         * reported as a simple root of value 0.
859                         */
860                         res.add(BigInteger.ZERO) ;
861                         res.add(BigInteger.ONE) ;
862                         return res ;
863                 }
864 
865                 /* multiply all coefs with the lcm() to get an integer polynomial
866                 * start with denominator of first non-zero coefficient.
867                 */
868                 BigInteger lcmDeno = a.elementAt(lowd).b ;
869                 for(int i=lowd+1; i < degree() ; i++)
870                         lcmDeno = BigIntegerMath.lcm(lcmDeno, a.elementAt(i).b ) ;
871 
872                 /* and eventually get the integer polynomial by ignoring the denominators
873                 */
874                 Vector<BigInteger> ipo = new Vector<BigInteger>() ;
875                 for(int i=0 ; i < a.size() ; i++)
876                 {
877                         BigInteger d = a.elementAt(i).a.multiply( lcmDeno).divide( a.elementAt(i).b)  ;
878                         ipo.add(d) ;
879                 }
880 
881                 BigIntegerPoly p = new BigIntegerPoly(ipo) ;
882                 /* collect the integer roots (multiple roots only once). Since we
883                 * removed the zero already above, cand does not contain zeros.
884                 */
885                 Vector<BigInteger> cand = p.iroots() ;
886                 for( int i =0 ; i < cand.size() ; i++)
887                 {
888                         final BigInteger r = cand.elementAt(i) ;
889                         int deg = p.rootDeg( r) ;
890                         res.add(r) ;
891                         res.add(new BigInteger(""+deg)) ;
892                 }
893 
894                 return res;
895         } /* iroots */
