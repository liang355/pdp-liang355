98          /** The Wigner 3jm symbol (j1,j2,j3,m1,m2,m3).
99          * All arguments of the function are the actual parameters multiplied by 2, so
100         * they all allow an integer representation.
101         * @param j1 integer representing 2*j1
102         * @param j2 integer representing 2*j2
103         * @param j3 integer representing 2*j3
104         * @param m1 integer representing 2*m1
105         * @param m2 integer representing 2*m2
106         * @param m3 integer representing 2*m3
107         * @return The value of the symbol. Zero if any of the triangular inequalities is
108         *  violated or some parameters are out of range.
109         * @since 2011-02-13
110         * @author Richard J. Mathar
111         */
112         static public BigSurd wigner3jm(int j1, int j2, int j3, int m1, int m2, int m3)
113         {
114                 Rational J1 = new Rational(j1,2) ;
115                 Rational J2 = new Rational(j2,2) ;
116                 Rational J3 = new Rational(j3,2) ;
117                 Rational M1 = new Rational(m1,2) ;
118                 Rational M2 = new Rational(m2,2) ;
119                 Rational M3 = new Rational(m3,2) ;
120                 return wigner3jm(J1,J2,J3,M1,M2,M3) ;
121         } /* wigner3jm */

123         /** Wigner 3jn symbol.
124         * For the 6j symbol, the input of the 3 lines is  "1 2 3 1 5 6", "4 5 3 4 2 6" "2j1+1 2j2+1 2j3+1 2l1+1 2l2+1 2l3+1"
125         * @param m1 The information on the number of angular momenta.
126         * @param t1 The list of one half of the triads, indexing j, whitespace separated
127         * @param t2 The list of the second half of the triads, indexing j, whitespace separated
128         * @param j The list of the integer values of the angular momenta.
129         *    They are actually the doubled j-values plus 1, whitespace separated. Only as many
130         *    as announced by the m1 parameter are used; trailing numbers are ignored.
131         * @see A. Bar-Shalom and M. Klapisch, <a href="http://dx.doi.org/10.1016/0010-4655(88)90192-0">NJGRAF...</a>, Comp. Phys Comm. 50 (3) (1988) 375
132         * @since 2011-02-13
133         * @since 2012-02-15 Upgraded return value to BigSurdVec
134         * @author Richard J. Mathar
135         */
136         static public BigSurdVec wigner3j(String m1, String t1, String t2, String j)
137         {
138                 /* The first number in the line "m" is the number of angular momenta.
139                 * The rest of the line is ignored.
140                 */
141                 Scanner s = new Scanner(m1) ;
142                 int m = s.nextInt() ;
143                 if ( m % 3 != 0 )
144                         throw new IllegalArgumentException("Angular momenta "+m+" not a multiple of three.") ;
145 
146                 /* Scan the numbers in the line "j". Excess numbers beyond what
147                 * has been announced in the "m" line are ignored.
148                 */
149                 int[] jvec = new int [m] ;
150                 int[] tvec = new int [2*m] ;
151 
152                 /* the third row contains positive 2j+1.
153                 */
154                 s = new Scanner(j) ;
155                 int ji=0 ;
156                 while( s.hasNextInt() && ji < m)
157                 {
158                         jvec[ji++] = s.nextInt() ;
159                         if ( jvec[ji-1] < 1 )
160                                 throw new IllegalArgumentException("Illegal value "+ jvec[ji-1] +" for 2j+1.") ;
161                 }
162 
163                 /* the first two rows contain signed values of indices into the j list
164                 */
165                 s = new Scanner(t1) ;
166                 int ti=0 ;
167                 while( s.hasNextInt())
168                         tvec[ti++] = s.nextInt() ;
169 
170                 s = new Scanner(t2) ;
171                 while( s.hasNextInt())
172                         tvec[ti++] = s.nextInt() ;
173 
174                 /* Basic sanity checks. All indices in the first two lines address
175                 * a number in the third line, and each index occurs exactly twice.
176                 */
177                 if ( ji % 3 != 0 )
178                         throw new IllegalArgumentException("j-count "+ji+" not a multiple of three.") ;
179                 if ( ti != 2*ji )
180                         throw new IllegalArgumentException("triad-count "+ti+" not twice j-count " + ji ) ;
181 
182                 int[] jfreq = new int[m] ;
183                 for(ji =0 ; ji < jfreq.length ; ji++)
184                         jfreq[ji] = 0 ;
185 
186                 /* maintain a 0-based index which shows where the j-value
187                 * has its first and second occurrence in the flattened list of triads.
188                 */
189                 int[][] jhash = new int[m][2] ;
190 
191                 for(ti =0 ; ti < 2*m ; ti++)
192                 {
193                         int t = tvec[ti] ;
194                         if ( t == 0 || Math.abs(t) > jvec.length )
195                                 throw new IllegalArgumentException("Triad index "+t+" out of bounds") ;
196                         if ( jfreq[Math.abs(t)-1] >= 2 )
197                                 throw new IllegalArgumentException("Node "+t+" referenced more than twice") ;
198                         jhash[Math.abs(t)-1][jfreq[Math.abs(t)-1]] = ti ;
199                         jfreq[Math.abs(t)-1]++ ;
200                 }
201 
202                 /* Move on from the 2j+1 values of the input to the j-values.
203                 * Subtract one and divide through 2.
204                 */
205                 Rational [] J = new Rational[jvec.length] ;
206                 for(ji=0 ; ji < jvec.length ; ji ++)
207                 {
208                         J[ji] = new Rational(jvec[ji]-1,2) ;
209                 }
210 
211                 /* Convert the 1-based indices to 0-based indices, loosing the sign information.
212                 */
213                 int [] triadidx = new int[tvec.length] ;
214                 for(ti = 0 ; ti < tvec.length; ti++)
215                         triadidx[ti] = Math.abs(tvec[ti])-1 ;
216 
217                 /* The M-values are all null (undetermined) at the start.
218                 */
219                 Rational [] M = new Rational[J.length] ;
220 
221                 return wigner3j(tvec,J,M,triadidx) ;
222         } /* wigner3j */

224         /** Wigner 3jn symbol.
225         * Computes sum_{mi} (-1)^(j1-m1+j2-m2+...) triad(triadidx[0..2])*triad(triadidx[3..5])*...
226         * where each factor is a Wigner-3jm symbol with each sign of m_i occurring once at the
227         * corresponding l-value.
228         * @param triadidx 0-based indices into the list of J
229         * @param J The list of J-values
230         * @param M The list of M-values associated with the J. This contains null where the parameter has
231         *   not yet been set by an outer loop.
232         * @since 2011-02-13
233         * @since 2012-02-15 Upgraded to return BigSurdVec
234         * @author Richard J. Mathar
235         */
236         static private BigSurdVec wigner3j(final int[] tvec, final Rational[] J, final Rational[] M,final int[] triadidx)
237         {
238                 /* The result of the computation. The sum over all m-combinations of the 
239                 * triads.
240                 */
241                 BigSurdVec res = new BigSurdVec() ;
242 
243                 /* First step is to monitor the triangular conditions on the J.
244                 * If at least one is violated, the result is zero. Loop over
245                 * the triads.
246                 */
247                 for(int t=0 ; t < triadidx.length ; t += 3)
248                 {
249                         /* Ensure |J[t]-J[t+1]| <= J[t+2] <= J[t]+J[t+1] */
250                         if ( J[triadidx[t]].subtract(J[triadidx[t+1]]).abs().compareTo( J[triadidx[t+2]]) > 0 )
251                                 return res ;
252                         if ( J[triadidx[t]].add(J[triadidx[t+1]]).compareTo( J[triadidx[t+2]]) < 0 )
253                                 return res ;
254                 }
255 
256                 /* the index of the preferred member of the triad list.
257                 * Preference given to those dangling in triads where alreaday two others are fixed,
258                 * then to members where at least one is fixed, then to smallest associated J-values.
259                 */
260                 int freeM = -1 ;
261                 int freeMrank = -1 ;
262                 for(int i=0 ; i < triadidx.length ; i++)
263                 {
264                         /* found an m-value which has not yet been summed over.
265                         */
266                         if ( M[triadidx[i]] == null)
267                         {
268                                 /* two cases: value is fixed implicitly because already two others values
269                                 * are set in the triad. or it is still to maintain its own explicit loop.
270                                 */
271                                 int triadn = i/3 ;
272                                 int triadr = i%3 ;
273                                 /* the neighbors in the triad have indices triadn*3+ (tiradr+1) mod 3 and triadn*3+(triadr+2) mod3
274                                 */
275                                 int nei1 = 3*triadn+(triadr+1)%3 ;
276                                 int nei2 = 3*triadn+(triadr+2)%3 ;
277 
278                                 /* found a candidate for which the two other values are already set.
279                                 */
280                                 if (M[triadidx[nei1]] != null && M[triadidx[nei2]] != null)
281                                 {
282                                         freeM = i ;
283                                         break;
284                                 }
285                                 else
286                                 {
287                                         /* rough work load estimator: basically (2J1+1)*(2J2+1)
288                                         */
289                                         Rational wt = J[triadidx[i]].multiply(2).add(1) ;
290                                         if (M[triadidx[nei1]] == null )
291                                                 wt = wt.multiply( J[triadidx[nei1]].multiply(2).add(1) );
292                                         if (M[triadidx[nei2]] == null )
293                                                 wt = wt.multiply( J[triadidx[nei2]].multiply(2).add(1) );
294                                         int thiswt = wt.intValue() ;
295                                         if ( freeM < 0 || thiswt < freeMrank)
296                                         {
297                                                 freeM = i ;
298                                                 freeMrank = thiswt ;
299                                         }
300                                 }
301                         }
302                 }
303 
304                 if ( freeM >= 0)
305                 {
306                         /* found an m-value which has not yet been summed over.
307                         */
308                         if ( M[triadidx[freeM]] == null)
309                         {
310                                 Rational[] childM = new Rational[M.length] ;
311                                 for(int ji=0 ; ji<M.length ; ji++)
312                                         if ( M[ji] != null)
313                                                 childM[ji] = M[ji] ;
314 
315                                 /* two cases: value is fixed implicitly because already two others values
316                                 * are set in the triad. or it is still to maintain its own explicit loop.
317                                 */
318                                 int triadn = freeM/3 ;
319                                 int triadr = freeM%3 ;
320                                 /* the neighbors in the triad have indices triadn*3+ (triadr+1) mod 3 and triadn*3+(triadr+2) mod3
321                                 */
322                                 int nei1 = 3*triadn+(triadr+1)%3 ;
323                                 int nei2 = 3*triadn+(triadr+2)%3 ;
324                                 if (M[triadidx[nei1]] == null || M[triadidx[nei2]] == null)
325                                 {
326                                         /* The J-value is J[triadidx[freeM]]. Loop from -J to +J, the allowed range.
327                                         */
328                                         Rational newm = J[triadidx[freeM]].negate() ;
329                                         while( newm.compareTo(J[triadidx[freeM]]) <= 0 )        
330                                         {
331                                                 childM[triadidx[freeM]] = tvec[freeM] >0 ? newm : newm.negate() ;
332                                                 res = res.add( wigner3j(tvec,J,childM,triadidx) ) ;
333                                                 newm = newm.add(Rational.ONE) ;
334                                         }
335                                 }
336                                 else
337                                 {
338                                         /* Set its value and the value at its companion j-value.
339                                         * Sum of the three m-values in the triad is to be zero for a non-zero contribution.
340                                         */
341                                         Rational m1 = M[triadidx[nei1]] ;
342                                         Rational m2 = M[triadidx[nei2]] ;
343                                         /* negate if these are the second occurrences of the J in the triads
344                                         */
345                                         if ( tvec[nei1] < 0)
346                                                 m1 = m1.negate() ;
347                                         if ( tvec[nei2] < 0)
348                                                 m2 = m2.negate() ;
349                                         /* m3 = -(m1+m2) */
350                                         Rational newm = tvec[freeM] > 0 ? m1.add(m2).negate() :m1.add(m2) ;
351                                         /* No contribution if the m-value enforced by the other two entries
352                                         * is outside the range -|J|..|J| enforced by its associated J-value. One could
353                                         * essentially remove this branching and let wigner3j() decide on this,
354                                         * but this is inefficient.
355                                         */
356                                         if ( newm.abs().compareTo( J[triadidx[freeM]] ) <= 0 )
357                                         {
358                                                 childM[triadidx[freeM]] = newm ;
359                                                 res = res.add( wigner3j(tvec,J,childM,triadidx) ) ;
360                                         }
361                                                 /* zero contribution if this m-value cannot be set to any
362                                                 * value compatible with the triangular conditions.
363                                                 */
364                                 }
365                                 return res ;
366                         }
367                 }
368 
369                 /* reached the bottom of the loop where all M-values are assigned.
370                 * Build the product over all Wigner-3jm values and the associated sign.
371                 */
372                 res = BigSurdVec.ONE ;
373                 for (int ji = 0 ; ji < triadidx.length ; ji += 3)
374                 {
375                         Rational m1 = M[triadidx[ji]] ;
376                         Rational m2 = M[triadidx[ji+1]] ;
377                         Rational m3 = M[triadidx[ji+2]] ;
378                         /* negate if these are associated with in-flowing vectors in the triads
379                         */
380                         if ( tvec[ji] < 0)
381                                 m1 = m1.negate() ;
382                         if ( tvec[ji+1] < 0)
383                                 m2 = m2.negate() ;
384                         if ( tvec[ji+2] < 0)
385                                 m3 = m3.negate() ;
386                         res = res.multiply( wigner3jm( J[triadidx[ji]], J[triadidx[ji+1]], J[triadidx[ji+2]], m1,m2,m3 ) );
387 
388                         /* if a partial product yields zero, the total product is zero, too, and
389                         * offers an early exit.
390                         */
391                         if ( res.signum() == 0 )
392                                 return BigSurdVec.ZERO ;
393                 }
394                 /* The overal sign is product_{J-Mpairs} (-1)^(J-M). This is an integer because all the J-M are integer.
395                 */
396                 Rational sig = new Rational() ;
397                 for(int ji=0 ; ji < J.length ; ji++)
398                         sig = sig.add(J[ji]).subtract(M[ji]) ;
399                 /* sign depends on the sum being even or odd. We assume that "sig" is integer and
400                 * look only at the numerator */
401                 if ( sig.a.abs().testBit(0) )
402                         res = res.negate() ;
403                 return res ;
404         } /* wigner3j */
405 
406         /** The Wigner 3jm symbol (j1,j2,j3,m1,m2,m3).
407         * Warning: there is no check that each argument is indeed half-integer.
408         * @param j1 integer or half-integer j1
409         * @param j2 integer or half-integer j2
410         * @param j3 integer or half-integer j3
411         * @param m1 integer or half-integer m1
412         * @param m2 integer or half-integer m2
413         * @param m3 integer or half-integer m3
414         * @return The value of the symbol. Zero if any of the triangular inequalities is
415         *  violated or some parameters are out of range.
416         * @since 2011-02-13
417         * @author Richard J. Mathar
418         */
419         static protected BigSurd wigner3jm(Rational j1, Rational j2, Rational j3, Rational m1, Rational m2, Rational m3)
420         {
421                 /* Check that m1+m2+m3 = 0 
422                 */
423                 if ( m1.add(m2).add(m3).signum() != 0 )
424                         return BigSurd.ZERO ;
425 
426                 /* Check that j1+j2+j3 is integer
427                 */
428                 if ( j1.add(j2).add(j3).isBigInteger() == false )
429                         return BigSurd.ZERO ;
430 
431                 /* Check that |j1-j2|<=j3 <= |j1+j2|
432                 */
433                 Rational j1m2 = j1.subtract(j2) ;
434                 if ( j1m2.abs().compareTo(j3) > 0 )
435                         return BigSurd.ZERO ;
436                 Rational j1p2 = j1.add(j2) ;
437                 if ( j1p2.abs().compareTo(j3) < 0 )
438                         return BigSurd.ZERO ;
439 
440                 /* Check that |m_i| <= j_i
441                 */
442                 if ( m1.abs().compareTo(j1) > 0 || m2.abs().compareTo(j2) > 0 || m3.abs().compareTo(j3) > 0 )
443                         return BigSurd.ZERO ;
444 
445                 /* Check that m_i-j_i are integer.
446                 */
447                 if ( ! m1.subtract(j1).isBigInteger() || ! m2.subtract(j2).isBigInteger() || ! m3.subtract(j3).isBigInteger() )
448                         return BigSurd.ZERO ;
449 
450                 /* (-)^(j1-j2-m3)*delta(-m3,m1+m2)*sqrt[ (j3+j1-j2)! (j3-j1+j2)! (j1+j2-j3)! /(j1+j2+j3+1)! 
451                 *                              *(j3-m)!*(j3+m)!(j1-m1)!*(j1+m1)!*(j2-m2)!*(j2+m2)!]
452                 *                               *sum_k (-1)^k/[k!(j1+j2-j3-k)!(j1-m1-k)!(j2+m2-k)!(j3-j2+m1+k)!)*(j3-j1-m2+k)!]
453                 */
454 
455                 /* It is tacitly assumed that all the major j_i, m_i values are in the integer range.
456                 * This is implicitly plausible since otherwise the execution times of the following loop over the k-values
457                 * would be immense.     
458                 */
459                 int j1j2jk = j1p2.subtract(j3).intValue() ;
460                 int j1m1k = j1.subtract(m1).intValue() ;
461                 int j2m2k = j2.add(m2).intValue() ;
462                 int jj2m1k = j3.subtract(j2).add(m1).intValue() ;
463                 int jj1m2k = j3.subtract(j1).subtract(m2).intValue() ;
464 
465                 int k = Math.max(0,-jj2m1k) ;
466                 k = Math.max(k,-jj1m2k) ;
467                 if ( k > 0 )
468                 {
469                         j1j2jk -= k ;
470                         j1m1k -= k ;
471                         j2m2k -= k ;
472                         jj2m1k += k ;
473                         jj1m2k += k ;
474                 }
475 
476                 Factorial f = new Factorial() ;
477                 Rational sumk = new Rational() ;
478                 while ( true)
479                 {
480                         BigInteger d = f.at(k)
481                                 .multiply(f.at(j1j2jk))
482                                 .multiply(f.at(j1m1k))
483                                 .multiply(f.at(j2m2k)) 
484                                 .multiply(f.at(jj2m1k))
485                                 .multiply(f.at(jj1m2k)) ;
486                         if ( k % 2 == 0)
487                                 sumk = sumk.add(new Rational(BigInteger.ONE,d)) ;
488                         else
489                                 sumk = sumk.subtract(new Rational(BigInteger.ONE,d)) ;
490                         j1j2jk-- ;
491                         j1m1k-- ;
492                         j2m2k-- ;
493                         jj2m1k++ ;
494                         jj1m2k++ ;
495                         if ( j1j2jk <0 || j1m1k < 0 || j2m2k < 0 )
496                                 break;
497                         k++ ;
498                 }
499                 /* sign factor (-1)^(j1-j2-m3)
500                 */
501                 if ( j1m2.subtract(m3).intValue() %2 != 0 )
502                         sumk = sumk.negate() ;
503 
504                 k = j1m2.add(j3).intValue() ;
505                 BigInteger s = f.at(k) ;
506                 k = j3.subtract(j1m2).intValue() ; s = s.multiply(f.at(k)) ;
507                 k = j1p2.subtract(j3).intValue() ; s = s.multiply(f.at(k)) ;
508                 k = j3.add(m3).intValue() ; s = s.multiply(f.at(k)) ;
509                 k = j3.subtract(m3).intValue() ; s = s.multiply(f.at(k)) ;
