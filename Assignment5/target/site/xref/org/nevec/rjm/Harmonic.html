7   /** Harmonic numbers.
8   * H(n) is the sum of the inverses of the integers from 1 to n.
9   * @since 2008-10-19
10  * @author Richard J. Mathar
11  */
12  public class Harmonic
13  {
14          /** ctor()
15          * Does nothing.
16          */
17          public Harmonic()
18          {
19          }
20  
21          /** The Harmonic number at the index specified
22          * @param n the index, non-negative.
23          * @return the H_1=1 for n=1, H_2=3/2 for n=2 etc.
24          *   For values of n less than 1, zero is returned.
25          */
26          public Rational at(int n)
27          {
28                  if ( n < 1)
29                          return(new Rational(0,1)) ;
30                  else
31                  {
32                          /* start with 1 as the result
33                          */
34                          Rational a = new Rational(1,1) ;
35  
36                          /* add 1/i for i=2..n
37                          */
38                          for( int i=2 ; i <=n ; i++)
39                                  a = a.add(new Rational(1,i)) ;
40                          return a ;
41                  }
42          }
43  } /* Harmonic */
