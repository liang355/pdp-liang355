7   /** Factored integers.
8   * This class contains a non-negative integer with the prime factor decomposition attached.
9   * @since 2006-08-14
10  * @since 2012-02-14 The internal representation contains the bases, and becomes sparser if few 
11  *    prime factors are present.
12  * @author Richard J. Mathar
13  */
14  public class Ifactor implements Cloneable, Comparable<Ifactor>
15  {
16          /**
17          * The standard representation of the number
18          */
19          public BigInteger n ;
20  
21          /*
22          * The bases and powers of the prime factorization.
23          * representation n = primeexp[0]^primeexp[1]*primeexp[2]^primeexp[3]*...
24          * The value 0 is represented by an empty vector, the value 1 by a vector of length 1
25          * with a single power of 0.
26          */
27          public Vector<Integer> primeexp ;
28  
29          final public static Ifactor ONE = new Ifactor(1) ;
30  
31          final public static Ifactor ZERO = new Ifactor(0) ;

33          /** Constructor given an integer.
34          * constructor with an ordinary integer
35          * @param number the standard representation of the integer
36          */
37          public Ifactor(int number)
38          {
39                  n = new BigInteger(""+number) ;
40                  primeexp = new Vector<Integer>() ;
41                  if( number > 1 )
42                  {
43                          int primindx = 0 ;
44                          Prime primes = new Prime() ;
45                          /* Test division against all primes.
46                          */
47                          while(number > 1)
48                          {
49                                  int ex=0 ;
50                                  /* primindx=0 refers to 2, =1 to 3, =2 to 5, =3 to 7 etc
51                                  */
52                                  int p = primes.at(primindx).intValue() ;
53                                  while( number % p == 0 )
54                                  {
55                                          ex++ ;
56                                          number /= p ;
57                                          if ( number == 1 )
58                                                  break ;
59                                  }
60                                  if ( ex > 0 )   
61                                  {
62                                          primeexp.add(new Integer(p)) ;
63                                          primeexp.add(new Integer(ex)) ;
64                                  }
65                                  primindx++ ;
66                          }
67                  }
68                  else if ( number == 1)
69                  {
70                          primeexp.add(new Integer(1)) ;
71                          primeexp.add(new Integer(0)) ;
72                  }
73          } /* Ifactor */

75          /** Constructor given a BigInteger .
76          * Constructor with an ordinary integer, calling a prime factor decomposition.
77          * @param number the BigInteger representation of the integer
78          */
79          public Ifactor(BigInteger number)
80          {
81                  n = number ;
82                  primeexp = new Vector<Integer>() ;
83                  if ( number.compareTo(BigInteger.ONE) == 0 )
84                  {
85                          primeexp.add(new Integer(1)) ;
86                          primeexp.add(new Integer(0)) ;
87                  }
88                  else
89                  {
90                          int primindx = 0 ;
91                          Prime primes = new Prime() ;
92                          /* Test for division against all primes.
93                          */
94                          while(number.compareTo(BigInteger.ONE) == 1)
95                          {
96                                  int ex=0 ;
97                                  BigInteger p = primes.at(primindx) ;
98                                  while( number.remainder(p).compareTo(BigInteger.ZERO) == 0 )
99                                  {
100                                         ex++ ;
101                                         number = number.divide(p) ;
102                                         if ( number.compareTo(BigInteger.ONE) == 0 )
103                                                 break ;
104                                 }
105                                 if ( ex > 0 )
106                                 {
107                                         primeexp.add(new Integer(p.intValue()) ) ;
108                                         primeexp.add(new Integer(ex) ) ;
109                                 }
110                                 primindx++ ;
111                         }
112                 }
113         } /* Ifactor */

115         /** Constructor given a list of exponents of the prime factor decomposition.
116         * @param pows the vector with the sorted list of exponents.
117         *  pows[0] is the exponent of 2, pows[1] the exponent of 3, pows[2] the exponent of 5 etc.
118         *  Note that this list does not include the primes, but assumes a continuous prime-smooth basis.
119         */
120         public Ifactor(Vector<Integer> pows)
121         {
122                 primeexp = new Vector<Integer>(2* pows.size()) ;
123                 if ( pows.size() > 0 )
124                 {
125                         n = BigInteger.ONE ;
126                         Prime primes = new Prime() ;
127                         /* Build the full number by the product of all powers of the primes.
128                         */
129                         for(int primindx=0 ; primindx < pows.size() ; primindx++)
130                         {
131                                 int ex= pows.elementAt(primindx).intValue() ;
132                                 final BigInteger p = primes.at(primindx) ;
133                                 n = n.multiply( p.pow(ex) ) ;
134                                 primeexp.add(new Integer(p.intValue()) ) ;
135                                 primeexp.add(new Integer(ex) ) ;
136                         }
137                 }
138                 else
139                         n = BigInteger.ZERO ;
140         } /* Ifactor */

230         /** Multiply with another positive integer.
231         * @param oth the second factor.
232         * @param type 0 to multiply, 1 for gcd, 2 for lcm
233         * @return the product, gcd or lcm of both numbers.
234         */
235         protected Ifactor multGcdLcm(final Ifactor oth, int type)
236         {
237                 Ifactor prod = new Ifactor(0) ;
238                 /* skip the case where 0*something =0, falling thru to the empty representation for 0
239                 */
240                 if( primeexp.size() != 0 && oth.primeexp.size() != 0)
241                 {
242                         /* Cases of 1 times something return something.
243                         * Cases of lcm(1, something) return something.
244                         * Cases of gcd(1, something) return 1.
245                         */
246                         if ( primeexp.firstElement().intValue() == 1 && type == 0)
247                                 return oth ;
248                         else if ( primeexp.firstElement().intValue() == 1 && type == 2)
249                                 return oth ;
250                         else if ( primeexp.firstElement().intValue() == 1 && type == 1)
251                                 return this ;
252                         else if ( oth.primeexp.firstElement().intValue() == 1 && type ==0)
253                                 return this ;
254                         else if ( oth.primeexp.firstElement().intValue() == 1 && type ==2)
255                                 return this ;
256                         else if ( oth.primeexp.firstElement().intValue() == 1 && type ==1)
257                                 return oth ;
258                         else
259                         {
260                                 int idxThis = 0 ;
261                                 int idxOth = 0 ;
262                                 switch(type)
263                                 {
264                                 case 0 :
265                                         prod.n = n.multiply(oth.n) ;
266                                         break;
267                                 case 1 :
268                                         prod.n = n.gcd(oth.n) ;
269                                         break;
270                                 case 2 :
271                                         /* the awkward way, lcm = product divided by gcd
272                                         */
273                                         prod.n = n.multiply(oth.n).divide( n.gcd(oth.n) ) ;
274                                         break;
275                                 }
276 
277                                 /* scan both representations left to right, increasing prime powers
278                                 */
279                                 while( idxOth < oth.primeexp.size()  || idxThis < primeexp.size() )
280                                 {
281                                         if ( idxOth >= oth.primeexp.size() )
282                                         {
283                                                 /* exhausted the list in oth.primeexp; copy over the remaining 'this'
284                                                 * if multiplying or lcm, discard if gcd.
285                                                 */
286                                                 if ( type == 0 || type == 2)
287                                                 {
288                                                         prod.primeexp.add( primeexp.elementAt(idxThis) ) ;
289                                                         prod.primeexp.add( primeexp.elementAt(idxThis+1) ) ;
290                                                 }
291                                                 idxThis += 2 ;
292                                         }
293                                         else if ( idxThis >= primeexp.size() )
294                                         {
295                                                 /* exhausted the list in primeexp; copy over the remaining 'oth'
296                                                 */
297                                                 if ( type == 0 || type == 2)
298                                                 {
299                                                         prod.primeexp.add( oth.primeexp.elementAt(idxOth) ) ;
300                                                         prod.primeexp.add( oth.primeexp.elementAt(idxOth+1) ) ;
301                                                 }
302                                                 idxOth += 2 ;
303                                         }
304                                         else
305                                         {
306                                                 Integer p ;
307                                                 int ex ;
308                                                 switch ( primeexp.elementAt(idxThis).compareTo(oth.primeexp.elementAt(idxOth) ) )
309                                                 {
310                                                 case 0 :
311                                                         /* same prime bases p in both factors */
312                                                         p = primeexp.elementAt(idxThis) ;
313                                                         switch(type)
314                                                         {
315                                                         case 0 :
316                                                                 /* product means adding exponents */
317                                                                 ex = primeexp.elementAt(idxThis+1).intValue() + 
318                                                                         oth.primeexp.elementAt(idxOth+1).intValue() ;
319                                                                 break;
320                                                         case 1 :
321                                                                 /* gcd means minimum of exponents */
322                                                                 ex = Math.min( primeexp.elementAt(idxThis+1).intValue() ,
323                                                                         oth.primeexp.elementAt(idxOth+1).intValue()) ;
324                                                                 break;
325                                                         default :
326                                                                 /* lcm means maximum of exponents */
327                                                                 ex = Math.max( primeexp.elementAt(idxThis+1).intValue() ,
328                                                                         oth.primeexp.elementAt(idxOth+1).intValue()) ;
329                                                                 break;
330                                                         }
331                                                         prod.primeexp.add( p ) ;
332                                                         prod.primeexp.add( new Integer(ex) ) ;
333                                                         idxOth += 2 ;
334                                                         idxThis += 2 ;
335                                                         break ;
336                                                 case 1:
337                                                         /* this prime base bigger than the other and taken later */
338                                                         if ( type == 0 || type == 2)
339                                                         {
340                                                                 prod.primeexp.add( oth.primeexp.elementAt(idxOth) ) ;
341                                                                 prod.primeexp.add( oth.primeexp.elementAt(idxOth+1) ) ;
342                                                         }
343                                                         idxOth += 2 ;
344                                                         break ;
345                                                 default:
346                                                         /* this prime base smaller than the other and taken now */
347                                                         if ( type == 0 || type == 2)
348                                                         {
349                                                                 prod.primeexp.add( primeexp.elementAt(idxThis) ) ;
350                                                                 prod.primeexp.add( primeexp.elementAt(idxThis+1) ) ;
351                                                         }
352                                                         idxThis += 2 ;
353                                                 }
354                                         }
355                                 }
356                         }
357                 }
358                 return prod ;
359         } /* Ifactor.multGcdLcm */

387         /** Exponentiation with a positive integer.
388         * @param exponent the non-negative exponent
389         * @return n^exponent. If exponent=0, the result is 1.
390         */
391         public Ifactor pow(final int exponent) throws ArithmeticException
392         {
393                 /* three simple cases first
394                 */
395                 if ( exponent < 0 )
396                         throw new ArithmeticException("Cannot raise "+ toString() + " to negative " + exponent) ;
397                 else if ( exponent == 0)
398                         return new Ifactor(1) ;
399                 else if ( exponent == 1)
400                         return this ;
401 
402                 /* general case, the vector with the prime factor powers, which are component-wise
403                 * exponentiation of the individual prime factor powers.
404                 */
405                 Ifactor pows = new Ifactor(0) ;
406                 for(int i=0 ; i < primeexp.size() ; i += 2)
407                 {
408                         Integer p = primeexp.elementAt(i) ;
409                         int ex = primeexp.elementAt(i+1).intValue() ;
410                         pows.primeexp.add( p ) ;
411                         pows.primeexp.add( new Integer(ex*exponent) ) ;
412                 }
413                 return pows ;
414         } /* Ifactor.pow */

416         /** Pulling the r-th root.
417         * @param r the positive or negative (nonzero) root.
418         * @return n^(1/r).
419         *   The return value falls into the Ifactor class if r is positive, but if r is negative
420         *   a Rational type is needed.
421         * @since 2009-05-18
422         */
423         public Rational root(final int r) throws ArithmeticException
424         {
425                 if ( r == 0 )
426                         throw new ArithmeticException("Cannot pull zeroth root of "+ toString()) ;
427                 else if ( r < 0 )
428                 {
429                         /* a^(-1/b)= 1/(a^(1/b))
430                         */
431                         final Rational invRoot = root(-r) ;
432                         return Rational.ONE.divide(invRoot) ;
433                 }
434                 else
435                 {
436                         BigInteger pows = BigInteger.ONE ;
437                         for(int i=0 ; i < primeexp.size() ; i += 2)
438                         {
439                                 /* all exponents must be multiples of r to succeed (that is, to
440                                 * stay in the range of rational results).
441                                 */
442                                 int ex = primeexp.elementAt(i+1).intValue() ;
443                                 if ( ex % r != 0 )
444                                         throw new ArithmeticException("Cannot pull "+ r+"th root of "+ toString()) ;
445 
446                                 pows.multiply( new BigInteger(""+primeexp.elementAt(i)).pow(ex/r) ) ;
447                         }
448                         /* convert result to a Rational; unfortunately this will loose the prime factorization */
449                         return new Rational(pows) ;
450                 }
451         } /* Ifactor.root */

454         /** The set of positive divisors.
455         * @return the vector of divisors of the absolute value, sorted.
456         * @since 2010-08-27
457         */
458         public Vector<BigInteger> divisors()
459         {
460                 /* Recursive approach: the divisors of p1^e1*p2^e2*..*py^ey*pz^ez are
461                 * the divisors that don't contain  the factor pz, and the
462                 * the divisors that contain any power of pz between 1 and up to ez multiplied
463                 * by 1 or by a product that contains the factors p1..py.
464                 */
465                 Vector<BigInteger> d=new Vector<BigInteger>() ;
466                 if ( n.compareTo(BigInteger.ZERO) == 0 )
467                         return d ;
468                 d.add(BigInteger.ONE) ;
469                 if ( n.compareTo(BigInteger.ONE) > 0 )
470                 {
471                         /* Computes sigmaIncopml(p1^e*p2^e2...*py^ey) */
472                         Ifactor dp = dropPrime() ;
473 
474                         /* get ez */
475                         final int ez = primeexp.lastElement().intValue() ;
476 
477                         Vector<BigInteger> partd = dp.divisors() ;
478 
479                         /* obtain pz by lookup in the prime list */
480                         final BigInteger pz = new BigInteger( primeexp.elementAt(primeexp.size()-2).toString()) ;
481 
482                         /* the output contains all products of the form partd[]*pz^ez, ez>0,
483                         * and with the exception of the 1, all these are appended.
484                         */
485                         for(int i =1 ; i < partd.size() ; i++)
486                                 d.add( partd.elementAt(i) ) ;
487                         for(int e =1 ; e <= ez ; e++)
488                         {
489                                 final BigInteger pzez = pz.pow(e) ;
490                                 for(int i =0 ; i < partd.size() ; i++)
491                                         d.add( partd.elementAt(i).multiply(pzez) ) ;
492                         }
493                 }
494                 Collections.sort(d) ;
495                 return d ;
496         } /* Ifactor.divisors */

506         /** Sum of the k-th powers of divisors of the number.
507         * @return the sum of all divisors of the number, 1^k+....+n^k.
508         */
509         public Ifactor sigma(int k)
510         {
511                 /* the question is whether keeping a factorization  is worth the effort
512                 * or whether one should simply multiply these to return a BigInteger...
513                 */
514                 if( n.compareTo(BigInteger.ONE) == 0 )
515                         return  ONE ;
516                 else if( n.compareTo(BigInteger.ZERO) == 0 )
517                         return  ZERO ;
518                 else
519                 {
520                         /* multiplicative: sigma_k(p^e) = [p^(k*(e+1))-1]/[p^k-1]
521                         * sigma_0(p^e) = e+1.
522                         */
523                         Ifactor resul = Ifactor.ONE ;
524                         for(int i=0 ; i < primeexp.size() ; i += 2)
525                         {
526                                 int ex = primeexp.elementAt(i+1).intValue() ;
527                                 if ( k == 0 )
528                                         resul = resul.multiply(ex+1) ;
529                                 else
530                                 {
531                                         Integer p = primeexp.elementAt(i) ;
532                                         BigInteger num = (new BigInteger(p.toString())).pow(k*(ex+1)).subtract(BigInteger.ONE) ;
533                                         BigInteger deno = (new BigInteger(p.toString())).pow(k).subtract(BigInteger.ONE) ;
534                                         /* This division is of course exact, no remainder
535                                         * The costly prime factorization is hidden here.
536                                         */
537                                         Ifactor f = new Ifactor(num.divide(deno)) ;
538                                         resul = resul.multiply(f) ;
539                                 }
540                         }
541                         return resul ; 
542                 }
543         } /* Ifactor.sigma */

545         /** Divide through the highest possible power of the highest prime.
546         * If the current number is the prime factor product p1^e1 * p2*e2* p3^e3*...*py^ey * pz^ez,
547         * the value returned has the final factor pz^ez eliminated, which gives
548         * p1^e1 * p2*e2* p3^e3*...*py^ey.
549         * @return the new integer obtained by removing the highest prime power.
550         *   If this here represents 0 or 1, it is returned without change.
551         * @since 2006-08-20
552         */
553         public Ifactor dropPrime()
554         {
555                 /* the cases n==1 or n ==0
556                 */
557                 if ( n.compareTo(BigInteger.ONE) <= 0 )
558                         return this ;
559 
560                 /* The cases n>1
561                 * Start empty. Copy all but the last factor over to the result
562                 * the vector with the new prime factor powers, which contain the
563                 * old prime factor powers up to but not including the last one.
564                 */
565                 Ifactor pows=new Ifactor(0) ;
566                 pows.n = BigInteger.ONE ;
567                 for(int i = 0 ; i < primeexp.size()-2 ; i += 2)
568                 {
569                         pows.primeexp.add( primeexp.elementAt(i)) ;
570                         pows.primeexp.add( primeexp.elementAt(i+1)) ;
571                         BigInteger p = new BigInteger( primeexp.elementAt(i).toString() ) ;
572                         int ex = primeexp.elementAt(i+1).intValue() ;
573                         pows.n = pows.n.multiply( p.pow(ex) ) ;
574                 }
575                 return pows ;
576         } /* Ifactor.dropPrime */

627         /** The Moebius function.
628         * 1 if n=1, else, if k is the number of distinct prime factors, return (-1)^k,
629         * else, if k has repeated prime factors, return 0.
630         * @return the moebius function.
631         */
632         public int moebius()
633         {
634                 if( n.compareTo(BigInteger.ONE) <= 0 )
635                         return 1 ;
636                 /* accumulate number of different primes in k */
637                 int k=1 ;
638                 for(int i=0 ; i < primeexp.size() ; i += 2)
639                 {
640                         final int e = primeexp.elementAt(i+1).intValue() ;
641                         if ( e > 1 )
642                                 return 0 ;
643                         else if ( e == 1)
644                                 /* accumulates (-1)^k */
645                                 k *= -1 ;
646                         
647                 }
648                 return( k ) ;
649         } /* Ifactor.moebius */
