7   /** A BigSurdVec represents an algebraic sum or differences of values
8   * which each term an instance of BigSurd.
9   * This mainly means that sums or differences of two BigSurd (or two BigSurdVec) can
10  * be represented (exactly) as a BigSurdVec.
11  * @since 2012-02-15
12  * @author Richard J. Mathar
13  */
14  public class BigSurdVec implements Comparable<BigSurdVec>
15  {
16          /** The value of zero.
17          */
18          static public BigSurdVec ZERO = new BigSurdVec() ;
19  
20          /** The value of one.
21          */
22          static public BigSurdVec ONE = new BigSurdVec(BigSurd.ONE) ;
23  
24          /** Internal representation: Each term as a single BigSurd.
25          * The value zero is represented by an empty vector.
26          */
27          Vector<BigSurd> terms ;
28  
29          /** Default ctor, which represents the zero.
30          * @since 2012-02-15
31          */
32          public BigSurdVec()
33          {
34                  terms = new Vector<BigSurd>() ;
35          } /* ctor */
36  
37          /** ctor given the value of a BigSurd.
38          * @param a The value to be represented by this vector.
39          * @since 2012-02-15
40          */
41          public BigSurdVec(BigSurd a)
42          {
43                  terms = new Vector<BigSurd>(1) ;
44                  terms.add(a) ;
45          } /* ctor */
46  
47          /** ctor given two values, which (when added) represent this number a+b.
48          * @param a The value to be represented by the first term of the vector.
49          * @param b The value to be represented by the second term of the vector.
50          * @since 2012-02-15
51          */
52          public BigSurdVec(BigSurd a, BigSurd b)
53          {
54                  terms = new Vector<BigSurd>(2) ;
55                  terms.add(a) ;
56                  terms.add(b) ;
57                  normalize() ;
58          } /* ctor */
59  
60          /** Combine terms that can be written as a single surd.
61          * This unites for example the terms sqrt(90) and sqrt(10) to 4*sqrt(10).
62          * @since 2012-02-15
63          */
64          protected void normalize()
65          {
66                  /* nothing to be done if at most one term
67                  */
68                  if ( terms.size() <=1 )
69                          return ;
70  
71                  Vector<BigSurd> newter = new Vector<BigSurd>() ;
72                  newter.add( terms.firstElement()) ;
73                  /* add j-th element to the existing vector and combine were possible
74                  */
75                  for(int j=1 ; j < terms.size() ; j++)
76                  {
77                          BigSurd todo = terms.elementAt(j) ;
78                          boolean merged = false ;
79                          for(int ex=0 ; ex < newter.size() ; ex++)
80                          {
81                                  BigSurd v = newter.elementAt(ex) ;
82                                  /* try to merge terms[j] and newter[ex].
83                                  * todo = r * v with r a rational number is needed. 
84                                  * Replaces v with v+todo = v*(1+r) if this reduction works.
85                                  */
86                                  BigSurd r = todo.divide(v) ;
87                                  if ( r.isRational() )
88                                  {
89                                          /* compute r+1 */
90                                          Rational newpref = r.toRational().add(1) ;
91                                          /* eliminate accidental zeros; overwrite with v*(1+r).
92                                          */
93                                          if ( newpref.compareTo( Rational.ZERO) == 0 )
94                                                  newter.removeElementAt(ex) ;
95                                          else
96                                          {
97                                                  v = v.multiply(newpref) ;
98                                                  newter.setElementAt(v,ex) ;
99                                          }
100                                         merged =true;
101                                         break ;
102                                 }
103                         }
104                         /* append if none of the existing elements matched
105                         */
106                         if ( ! merged)
107                                 newter.add(todo) ;
108                 }
109 
110                 /* overwrite old version */
111                 terms = newter ;
112         } /* normalize */
113                 
114         /** Compare algebraic value with oth.
115         * Returns -1, 0 or +1 depending on whether this is smaller, equal to or larger than oth.
116         * @param oth The value with which this is to be compared.
117         * @return 0 or +-1.
118         * @since 2012-02-15
119         */
120         public int compareTo(BigSurdVec oth)
121         {
122                 final BigSurdVec diff = this.subtract(oth) ;
123                 return diff.signum() ;
124         } /* compareTo */
125         
126         /** Sign function.
127         * Returns -1, 0 or +1 depending on whether this is smaller, equal to or larger than zero.
128         * @return 0 or +-1.
129         * @since 2012-02-15
130         */
131         public int signum()
132         {
133                 /* the case of zero is unique, because no (reduced) vector of surds other than
134                 * the one element 0 itself can add/subtract to zero.
135                 */
136                 if ( terms.size() == 0 )
137                         return 0 ;
138 
139                 /* if there is one term: forward to the signum function of BigSurd
140                 */
141                 if ( terms.size() == 1 )
142                         return terms.firstElement().signum() ;
143 
144                 /* if all terms have a common sign: take that one
145                 * offsig is the index of the first "offending" term in the sense that its
146                 * sign doese not agree with the term[0].
147                 */
148                 int sig0 = terms.elementAt(0).signum() ;
149                 int offsig = 1;
150                 for( ; offsig < terms.size() ; offsig++)
151                         if ( terms.elementAt(offsig).signum() != sig0)
152                                 break;
153                 if ( offsig >= terms.size() )
154                         return sig0 ;
155 
156                 /* if there are two terms (now known to have different sign):
157                 * forward to the comparison of the two elements as BigSurds
158                 */
159                 if ( terms.size() == 2 )
160                         return terms.elementAt(0).compareTo( terms.elementAt(1).negate() ) ;
161 
162                 /* if there are three terms, move the one with the offending sign to the
163                 * other side and square both sides (which looses the sign) to remove all but one surds.
164                 * The difference of the squared sides contains at most two terms, which reduces to the
165                 * case above.
166                 *  t(0)+t(offbar) <> -t(offs)
167                 */
168                 if ( terms.size() == 3 )
169                 {
170                         BigSurdVec lhs ;
171                         if ( offsig == 2)
172                                 lhs = new BigSurdVec( terms.elementAt(0), terms.elementAt(1)) ;
173                         else
174                                 lhs = new BigSurdVec( terms.elementAt(0), terms.elementAt(2)) ;
175                         lhs = lhs.sqr() ;
176                         BigSurd rhs = new BigSurd(terms.elementAt(offsig).sqr(),Rational.ONE) ;
177                         if ( lhs.compareTo(lhs) > 0 )
178                                 /* dominating sign was t(0)+t(offbar)
179                                 */
180                                 return terms.elementAt(0).signum() ;
181                         else
182                                 return terms.elementAt(offsig).signum() ;
183                 }
184 
185                 /* for a larger number of terms: take a floating point representation with a small
186                 * but correct number of digits, and resume with the sign of that one.
187                 */
188                 return (floatValue() > 0. ) ? 1 : -1 ;
189 
190         } /* signum */

192         /** Construct an approximate floating point representation
193         * @param mc The intended accuracy of the result.
194         * @return A truncated version with the precision described by mc
195         */
196         public BigDecimal BigDecimalValue(MathContext mc)
197         {
198                 /* simple cases with one term forwarded to the BigSurd class
199                 */
200                 if ( terms.size() == 0 )
201                         return BigDecimal.ZERO ;
202                 else if ( terms.size() == 1 )
203                 {
204                         return terms.firstElement().BigDecimalValue(mc) ;
205                 }
206 
207                 /* To reduce cancellation errors, loop over increasing local precision
208                 * until we are stable to the required result. Keep the old (less precise)
209                 * estimate in res[0], and the newer, more precise in res[1].
210                 */
211                 BigDecimal [] res = new BigDecimal[2] ;
212                 res[0] = BigDecimal.ZERO ;
213                 for(int addpr =1 ; ; addpr += 3)
214                 {
215                         MathContext locmc = new MathContext(mc.getPrecision()+addpr,mc.getRoundingMode()) ;
216                         res[1] = BigDecimal.ZERO ;
217                         for( BigSurd j : terms)
218                                 res[1] = BigDecimalMath.addRound(res[1], j.BigDecimalValue(locmc) ) ;
219                         if ( addpr > 1 )
220                         {
221                                 BigDecimal err = res[1].subtract(res[0]).abs() ;
222                                 int prec = BigDecimalMath.err2prec(res[1],err) ;
223                                 if ( prec > mc.getPrecision() )
224                                         break ;
225                         }
226                         res[0] = res[1] ;
227                 }
228                 return BigDecimalMath.scalePrec(res[1],mc) ;
229         
230         } /* BigDecimalValue */

250         /** Add two vectors algebraically.
251         * @param val The value to be added to this.
252         * @return The new value representing this+val.
253         */
254         public BigSurdVec add(final BigSurdVec val)
255         {
256                 BigSurdVec sum = new BigSurdVec() ;
257                 /* concatenate the vectors and eliminate common overlaps
258                 */
259                 sum.terms.addAll( terms) ;
260                 sum.terms.addAll( val.terms) ;
261                 sum.normalize() ;
262                 return sum ;
263         } /* add */
264 
265         /** Add two vectors algebraically.
266         * @param val The value to be added to this.
267         * @return The new value representing this+val.
268         */
269         public BigSurdVec add(final BigSurd val)
270         {
271                 BigSurdVec sum = new BigSurdVec() ;
272                 /* concatenate the vectors and eliminate common overlaps
273                 */
274                 sum.terms.addAll( terms) ;
275                 sum.terms.add( val) ;
276                 sum.normalize() ;
277                 return sum ;
278         } /* add */

280         /** Subtract another number.
281         * @param val The value to be subtracted from this.
282         * @return The new value representing this-val.
283         */
284         public BigSurdVec subtract(final BigSurdVec val)
285         {
286                 BigSurdVec sum = new BigSurdVec() ;
287                 /* concatenate the vectors and eliminate common overlaps
288                 */
289                 sum.terms.addAll( terms) ;
290                 for ( BigSurd s : val.terms)
291                         sum.terms.add( s.negate() ) ;
292                 sum.normalize() ;
293                 return sum ;
294         } /* subtract */

297         * @param val The value to be subtracted from this.
298         * @return The new value representing this-val.
299         */
300         public BigSurdVec subtract(final BigSurd val)
301         {
302                 BigSurdVec sum = new BigSurdVec() ;
303                 /* concatenate the vectors and eliminate common overlaps
304                 */
305                 sum.terms.addAll( terms) ;
306                 sum.terms.add( val.negate() ) ;
307                 sum.normalize() ;
308                 return sum ;
309         } /* subtract */

311         /** Compute the negative.
312         * @return -this.
313         * @since 2012-02-15
314         */
315         public BigSurdVec negate()
316         {
317                 /* accumulate the negated elements of term one by one
318                 */
319                 BigSurdVec resul = new BigSurdVec() ;
320                 for( BigSurd s : terms)
321                         resul.terms.add( s.negate()) ;
322                 /* no normalization step here, because the negation of all terms
323                 * does not introduce new common factors
324                 */
325                 return resul ;
326         } /* negate */

328         /** Compute the square.
329         * @return this value squared.
330         * @since 2012-02-15
331         */
332         public BigSurdVec sqr()
333         {
334                 /* Binomial expansion. First the sum of the terms squared,
335                 * then 2 times the mixed products.
336                 */
337                 BigSurdVec resul = new BigSurdVec() ;
338                 for( int i=0 ; i < terms.size() ; i++)
339                         resul.terms.add( new BigSurd(terms.elementAt(i).sqr(),Rational.ONE) ) ;
340                 for( int i=0 ; i < terms.size()-1 ; i++)
341                 for( int j= i+1 ; j < terms.size() ; j++)
342                         resul.terms.add( terms.elementAt(i).multiply(terms.elementAt(j)).multiply(2) ) ;
343                 resul.normalize() ;
344                 return resul ;
345         } /* sqr */

347         /** Multiply by another square root.
348         * @param val a second number of this type.
349         * @return the product of this with the val.
350         * @since 2011-02-12
351         */
352         public BigSurdVec multiply(final BigSurd val)
353         {
354                 BigSurdVec resul = new BigSurdVec() ;
355                 for( BigSurd s : terms)
356                         resul.terms.add( s.multiply(val) )  ;
357                 resul.normalize() ;
358                 return resul ;
359         } /* multiply */

361         /** Return a string in the format (number/denom)*()^(1/2).
362         * If the discriminant equals 1, print just the prefactor.
363         * @return the human-readable version in base 10
364         * @since 2012-02-16
365         */
366         public String toString()
367         {
368                 /* simple cases with one term forwarded to the BigSurd class
369                 */
370                 if ( terms.size() == 0 )
371                         return new String("0") ;
372                 else 
373                 {
374                         String s = new String() ;
375                         for(int t=0 ; t< terms.size() ; t++)
376                         {
377                                 BigSurd bs = terms.elementAt(t) ;
378                                 if ( bs.signum() > 0 )
379                                         s += "+" ;
380                                 s += bs.toString() ;
381                         }
382                         return s ;
383                 }
384         } /* toString */
