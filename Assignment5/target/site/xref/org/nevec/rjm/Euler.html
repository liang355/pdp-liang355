4   import java.util.* ;

7   /** Euler numbers
8   * @see <a href="http://oeis.org/A000364">A000364</a> in the OEIS.
9   * @since 2008-10-30
10  * @author Richard J. Mathar
11  */
12  public class Euler
13  {
14          /*
15          * The list of all Euler numbers as a vector, n=0,2,4,....
16          */
17          static protected Vector<BigInteger> a = new Vector<BigInteger>() ;
18  
19          /** Ctor(). Fill the hash list initially with E_0 to E_3.
20          */
21          public Euler()
22          {
23                  if ( a.size() == 0 )
24                  {
25                          a.add(BigInteger.ONE) ;
26                          a.add(BigInteger.ONE) ;
27                          a.add(new BigInteger("5")) ;
28                          a.add(new BigInteger("61")) ;
29                  }
30          }
31  
32          /** Compute a coefficient in the internal table.
33          * @param n the zero-based index of the coefficient. n=0 for the E_0 term. 
34          */
35          protected void set(final int n)
36          {
37                  while ( n >= a.size())
38                  {
39                          BigInteger val = BigInteger.ZERO ;
40                          boolean sigPos = true; 
41                          int thisn = a.size() ;
42                          for(int i= thisn-1 ; i > 0 ; i--)
43                          {
44                                  BigInteger f = new BigInteger(""+ a.elementAt(i).toString() ) ;
45                                  f = f.multiply( BigIntegerMath.binomial(2*thisn,2*i) );
46                                  if ( sigPos )
47                                          val = val.add(f) ;
48                                  else
49                                          val = val.subtract(f) ;
50                                  sigPos = ! sigPos ;
51                          }
52                          if ( thisn % 2 ==0 )
53                                  val = val.subtract(BigInteger.ONE) ;
54                          else
55                                  val = val.add(BigInteger.ONE) ;
56                          a.add(val) ;
57                  }
58          }
59  
60          /** The Euler number at the index provided.
61          * @param n the index, non-negative.
62          * @return the E_0=E_1=1 , E_2=5, E_3=61 etc
63          */
64          public BigInteger at(int n)
65          {
66                  set(n) ;
67                  return(a.elementAt(n)) ;
68          }
69  
70  } /* Euler */
