126         * @since 2010-08-27

128         public void compute()
129         {
130                 String tr = inpGtria.getText() ;
131                 String[] trias = new String[4] ;
132 
133                 /* Read the trias configuration from inpGtria into trias[0..2], skipping lines
134                 * that start with a hash mark.
135                 */
136                 Scanner s = new Scanner(tr) ;
137                 for(int l=0 ; l < 3 ; )
138                 {
139                         try
140                         {
141                                 trias[l] = s.nextLine().trim() ;
142                                 if ( ! trias[l].startsWith("#") )
143                                         l++ ;
144                         }
145                         catch(Exception e)
146                         {
147                                 outG.setText("ERROR: less than 3 lines in the triad definition") ;
148                                 return ;
149                         }
150                 }
151                 /* Read the J values from inpGjval into trias[3] in a loop
152                 */
153                 String j = inpGjval.getText() ;
154                 s = new Scanner(j) ;
155                 while ( true)
156                 {
157                         try
158                         {
159                                 trias[3] = s.nextLine().trim() ;
160                         }
161                         catch(Exception e)
162                         {
163                                 return ;
164                         }
165                         if ( ! trias[3].startsWith("#") )
166                         {
167                                 try {
168                                         BigSurdVec w = Wigner3j.wigner3j(trias[0],trias[1],trias[2],trias[3]) ;
169                                         outG.append(w.toString()+ " = " + w.doubleValue()) ;
170                                 }
171                                 catch (Exception e)
172                                 {
173                                         outG.append(e.toString()) ;
174                                         e.printStackTrace() ;
175                                 }
176                                 outG.append(" # J = ") ;
177                                 Scanner num = new Scanner(trias[3]) ;
178                                 while( num.hasNextInt() )
179                                 {
180                                         int twoj1 = num.nextInt() ;
181                                         Rational jfrac = new Rational(twoj1-1,2) ;
182                                         outG.append(jfrac.toString() + " ") ;
183                                 }
184                                 outG.append("\n") ;
185                         }
186                 }
187         } /* compute */
